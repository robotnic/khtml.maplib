  3 khtml.maplib.overlay.FeatureCollection = function() {
  4 	this.type="Feature";
  5 	this.className=new Object;
  6 	this.realLayer=false;  //true -> only one vector element
  7 	this.properties=new Object;
  8 	this.geometry=new Object;
  9 	this.geometry.type="FeatureCollection";
 10 	this.features=new Array();
 11 	this.overlayDiv=document.createElement("div");
 12 	this.overlayDiv.setAttribute("kazalabas","eluag");
 13 	this.style=new Object;
 14 	//this.backend="svg";
 15 	//this.className=this.overlayDiv.className;
 16 //	this.overlaySVG=new khtml.maplib.overlay.Vector();
 17 	this.init=function(owner){
 18 		this.owner=owner;
 19 		if(owner instanceof khtml.maplib.base.Map){
 20 			this.map=owner;
 21 			if(!this.overlayDiv.parentNode){
 22 			this.map.overlayDiv.appendChild(this.overlayDiv);	
 23 			}
 24 		}else{
 25 			if(owner.map){
 26 				this.map=owner.map;
 27 				if(!this.overlayDiv.parentNode){
 28 				this.owner.overlayDiv.appendChild(this.overlayDiv);
 29 				}
 30 			}else{
 31 				return;
 32 			}
 33 		}
 34 		//if(!this.owner.backend)this.backend=this.owner.backend;
 35 		//if(!this.backend)this.backend="canvas";
 36 		if(this.owner && this.owner.backend){
 37 			this.backend=this.owner.backend;
 38 		}
 39 		this.vectorEl=this.makeVectorLayer(this.backend);
 40 		//for(var i=0;i<this.features.length;i++){
 41 		for(var i in this.features){
 42 			if(!this.features[i].map){
 43 				var el=this.features[i];		
 44 
 45 				if(el.type=="Marker"){
 46 					el.init(this);
 47 				}else{
 48 					if(el.geometry.type=="FeatureCollection" || el.geometry.type=="GroundOverlay"){
 49 						el.init(this);
 50 					}else{
 51 						this.vectorLayer.createPolyline(el);
 52 						this.vectorLayer.render(el);
 53 					}
 54 				}
 55 				this.extendBBox(el);
 56 			}
 57 		}			
 58 		if(this.owner && this.owner.extendBBox){
 59 			this.owner.extendBBox(this);
 60 		}
 61 	}
 62 	this.makeVectorLayer=function(){
 63 		if(this.vectorLayer)return;
 64 		if(this.owner){
 65 				//this.vectorLayer=this.owner.vectorLayer;
 66 				this.vectorLayer=new khtml.maplib.overlay.Vector(this.backend);
 67 				if(this.map){
 68 					if(this.owner.vectorLayer && !this.realLayer){
 69 						vectorEl=this.vectorLayer.init(this,this.owner);  //g layer for css and fast rendering
 70 					}else{
 71 						vectorEl=this.vectorLayer.init(this); //svg, canvas, vml element
 72 					}
 73 					for(var p in this.style){
 74 						var prop=this.style[p];
 75 						if(this.vectorLayer.backend!="canvas"){
 76 							vectorEl.style[p]=prop;
 77 						}
 78 					}
 79 					if(this.vectorLayer.backend!="canvas"){
 80 						this.style=vectorEl.style;  //inherit css
 81 					}
 82 				}
 83 		}
 84 		return vectorEl;
 85 	}
 86 	this.renderbackend=function(backend){
 87 		if(this.vectorLayer){
 88 			this.vectorLayer.remove();
 89 		}
 90 		this.owner.backend=backend;
 91 		/*
 92 		for(var i=0; i<this.features.length;i++){	
 93 			if(this.features[i].renderbackend){
 94 				this.features[i].renderbackend(backend);	
 95 			}
 96 		}
 97 		*/
 98 		this.init(this.owner);
 99 	}
100 	this.oldStyleDisplay=null;
101 	this.render=function(){
102 		if(!this.overlayDiv.parentNode){
103 			//this.owner.overlayDiv.appendChild(this.overlayDiv);
104 		}
105 		if(this.vectorLayer.backend=="canvas" && this.realLayer){
106 			this.vectorLayer.clear();
107 		}
108 		if(this.vectorLayer.backend=="vml" && this.realLayer){
109 			this.vectorLayer.clear();
110 		}
111 		for(var i=0; i<this.features.length;i++){	
112 			if(!this.features[i]) continue;
113 			if(this.features[i].geometry.type=="Point"){
114 				this.features[i].render();
115 			}
116 			if(this.features[i].geometry.type=="FeatureCollection"){
117 				//this.features[i].render();	
118 			}	
119 			if(this.features[i].geometry.type=="GroundOverlay"){
120 				this.features[i].render();	
121 			}	
122 		}	
123 		if(this.vectorLayer){
124 			this.vectorLayer.render();
125 		}
126 		if(this.oldStyleDisplay!=null){
127 			this.style.display=this.oldStyleDisplay;
128 		}
129 		this.oldStyleDisplay=null;
130 
131 	}	
132 	this.clear=function(){
133 //		this.owner.overlayDiv.removeChild(this.overlayDiv);
134 		if(this.oldStyleDisplay==null){
135 			this.oldStyleDisplay=this.style.display;
136 			this.style.display="none";
137 		}
138 		for(var i=0; i< this.features.length;i++){
139 			if(this.features[i].type=="Marker"){
140 				this.features[i].clear();
141 			}
142 			if(this.features[i].geometry.type=="FeatureCollection" && !this.realLayer){
143 				this.features[i].clear();	
144 			}	
145 		}
146 	}
147 	this.removeChild=function(el){
148 		var found=false;
149 		for(var f in this.features){
150 			if(this.features[f]==el){
151 				this.features.splice(f,1);	
152 				found=true;
153 			}
154 		}
155 		if(found){
156 			delete el.owner;
157 			if(this.owner){
158 				this.render();
159 			}
160 			return true;
161 		}else{
162 			return false;
163 		}
164 	}
165 	this.appendChild=function(el){
166 		if(el.owner){
167 			var r=el.owner.removeChild(el);
168 		}
169 		this.features.push(el);
170 		if(!el.geometry)return;
171 		el.owner=this;
172 		if(el.geometry.type!=="Point" && el.geometry.type!=="FeatureCollection"&& el.geometry.type!=="GroundOverlay"){
173 			if(this.vectorLayer){  //if owner is initialized
174 				this.vectorLayer.createPolyline(el);
175 				this.vectorLayer.render(el);
176 			}
177 		}else{
178 			if(el.owner.map){
179 				el.init(this);
180 				el.render();
181 			}else{
182 			}
183 		}
184 		
185 		this.extendBBox(el);
186 		if(this.owner && this.owner.extendBBox){
187 			this.owner.extendBBox(el);
188 		}
189 	}
190 	this.extendBBox=function(el){
191 		if(el.point){
192 			var bbox=el.point;
193 		}else{
194 			var bbox=el.bbox;
195 		}
196 		if(!bbox)return;
197 		this.bbox=khtml.maplib.base.helpers.extendBBox(this.bbox,bbox);
198 		return;
199 
200 
201 		if(this.bbox){
202 			this.bbox=khtml.maplib.base.helpers.extendBBox(this.bbox,el);
203 		}else{
204 			if(el.bbox){
205 				this.bbox=el.bbox;
206 			}else{
207 				if(el.geometry.coordinates instanceof khtml.maplib.LatLng){  //node, Marker, Point
208 					this.bbox=new khtml.maplib.geometry.Bounds(el.geometry.coordinates,el.geometry.coordinates);
209 				}
210 			}
211 		}
212 	}
213 }
