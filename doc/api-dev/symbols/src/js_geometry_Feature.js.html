1 /**
  2 	Feature
  3 	
  4 	@example Call like this:
  5 	<pre>
  6 	var geometry={
  7 		type:"LineString",
  8 		coordinates:[[10,01],[20,20]],
  9 	}
 10 	
 11 	khtml.maplib.geometry.Feature(geometry);
 12 	
 13 	or
 14 	
 15 	khtml.maplib.geometry.Feature({type:"LineString",coordinates:[[10,01],[20,20]]});
 16 	</pre>
 17  * @class
 18 * @see Example: <a href="../../../examples/geometry/feature.html">Vectors</a>
 19 */
 20 
 21 
 22 khtml.maplib.geometry.Feature = function(geometry) {
 23 	this.type="Feature";
 24 	this.geometry=new Object;
 25 	this.geometry.coordinates=new Array();
 26 	this.properties=new Object();
 27 	this.events = new Object;
 28 	this.style = new Object;
 29 	this.className = new Object;
 30 
 31 	this.text = new Object;
 32 	this.text.style = new Object;
 33 	this.text.className = new Object;
 34 	this.className.baseVal = "khtml_vector_line";
 35 	if(typeof(geometry)=="object"){
 36 		if(geometry.type){
 37 			if(geometry.type=="Feature"){
 38 				this.geometry.type=geometry.geometry.type;
 39 				this.geometry.coordinates=khtml.maplib.base.helpers.parseLine(geometry.geometry.coordinates);
 40 				if(geometry.className){
 41 					this.className.baseVal=geometry.className.baseVal;
 42 				};
 43 				if(geometry.geometry.type=="FeatureCollection"){
 44 					var fc=new khtml.maplib.overlay.FeatureCollection();
 45 
 46 					for(var i=0;i<geometry.features.length;i++){
 47 						var feature=geometry.features[i];
 48 						fc.appendChild(new khtml.maplib.geometry.Feature(feature));
 49 					}
 50 					return fc;	
 51 				}
 52 			}else{
 53 				this.geometry.type=geometry.type;
 54 				if(geometry.coordinates){
 55 					this.geometry.coordinates=khtml.maplib.base.helpers.parseLine(geometry.coordinates);
 56 					return;
 57 				}
 58 				if(geometry.type=="FeatureCollection"){
 59                                         var fc=new khtml.maplib.overlay.FeatureCollection();
 60 					if(geometry.feature){
 61 						for(var i=0;i<geometry.features.length;i++){
 62 							var feature=geometry.features[i];
 63 							fc.appendChild(new khtml.maplib.geometry.Feature(feature));
 64 						}
 65 					}
 66                                         return fc
 67 				}
 68 			}
 69 		}
 70 	}
 71 	if(typeof(geometry)=="string"){ //ok why not
 72 		this.geometry.type=geometry;
 73 	}
 74 	/**
 75 	depressiated
 76 	*/
 77 	this.setPoints=function(points){
 78 		this.geometry.coordinates=khtml.maplib.base.helpers.parseLine(points);
 79 	}
 80 	/**
 81 	Serialize overlays to json string
 82 	*/
 83 	this.geojson=function(){
 84 		return khtml.maplib.base.helpers.stringify(this);
 85 	}
 86 	/**
 87 	not implemented
 88 	*/
 89 	this.cloneNode=function(){
 90 		return khtml.maplib.base.helpers.cloneObject(this);
 91 	}
 92 
 93 }
