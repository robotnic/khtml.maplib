 31 /**
 32  * Vector, internaly used
 33  * This Object represents one FeatureCollection with man Vectors. Markers are not handles with this Class
 34  * 
 35  * What can it do for you?
 36  * o It does the rendering on SVG, Canvas, VML Vector Backend
 37  * o Does the CSS based Styling. For SVG this is done by browser. For other Backends the properties like
 38  *   fill, stroke, stroke-width, opacity, fill-opacity, stroke-opacity, display, or translated to native
 39  *   styling syntax. Also CSS hierarchy is done here.
 40  *   CSS converting needs lots of CPU Power. More in Memory-caching could leed to better performance
 41  * o Speedy and lag-free userinterface is important. This Object should not freece the browser.
 42  *
 43  * @class
 44 */

 82 		if(backend){

 97 	this.createVirtualVectorElement=function(ownerVectorElement){
 98 		switch(this.backend){
 99 			case "canvas":
100 			case "vml":
101 				//not acceleration yes
102 				return ownerVectorElement;
103 				break;
104 			case "svg":
105 				var g=document.createElementNS("http://www.w3.org/2000/svg","g");
106 				if(this.owner.className && this.owner.className.baseVal){
107 					g.className.baseVal=this.owner.className.baseVal;
108 				}
109 				this.owner.className=g.className;
110 				return g;
111 				break;
112 			default:
113 				alert("unknown backend: "+this.backend);
114 		}
115 	}

137 				stl.addRule("v\\:group", "behavior: url(#default#VML);");
138 				stl.addRule("v\\:polyline", "behavior: url(#default#VML);");
139 				stl.addRule("v\\:stroke", "behavior: url(#default#VML);");
140 				stl.addRule("v\\:fill", "behavior: url(#default#VML);");
141 				stl.addRule("v\\:shape", "behavior: url(#default#VML);display:inline-block");
142 				stl.addRule("v\\:path", "behavior: url(#default#VML);");

161 		vectorEl.style.position = "absolute";
162 		vectorEl.style.top = "0";
163 		vectorEl.style.left = "0";
164 		return vectorEl;
165 	}

175                         this.themap=owner;
176                 }else{
177                         this.themap=owner.map;

184 			if(this.backend=="vml"){
185 				vectorEl=fcWithVectorElement.vectorEl;
186 			}
187 			if(this.backend=="canvas"){
188 				this.ctx=fcWithVectorElement.vectorLayer.ctx;
189 			}
190

192 			var vectorEl = this.createVectorElement(this.themap);
193 			this.owner.overlayDiv.appendChild(vectorEl);
194 		}
195 		return vectorEl;

201 	this.createPolyline = function(polyline) {
202 		polyline.tags = new Array;
203 		if(!polyline.close){
204 			polyline.close = false;
205 		}
206 		polyline.id = null;
207 		if(!polyline.geometry.coordinates){
208 				polyline.geometry.coordinates=new Array();
209 		}	
210 		
211 		polyline.calculateCenter = function() {
212 			var sumLat=0.0; // float!
213 			var sumLng=0.0; // float!
214 			for(var i = 0; i < polyline.points.length; i++) {
215 				sumLat += polyline.points[i].lat();
216 				sumLng += polyline.points[i].lng();
217 			}
218 			return new khtml.maplib.LatLng(sumLat/polyline.points.length/*lat*/, sumLng/polyline.points.length/*lng*/);
219 		}
220 		var that=this;
221 		polyline.repairBBox=function(force){
222 			if(force || polyline.geometry.coordinates.length != polyline.length){
223 				polyline.bbox=that.makeBounds(polyline.geometry.coordinates);
224 				polyline.length=polyline.geometry.coordinates.length;
225 			}
226 		}
227 		polyline.repairBBox(true);
228 		//render a single vector element
229 		polyline.render=function(){
230 			if(polyline && polyline.owner){  //workaround for removeChild bug
231 				that.render(polyline);
232 			}
233 		}
234 		polyline.bbox=this.makeBounds(polyline.geometry.coordinates);  //do I need that?
235 		polyline.clear=function(){
236 			//VML missing
237 			if(that.backend=="svg"){
238 				if(polyline.path && polyline.path.parentNode){
239 					polyline.path.parentNode.removeChild(polyline.path);
240 				}
241 			}
242 			//that.clear();
243 		}
244 		/* remove completely */
245 		polyline.destroy=function(){
246 			polyline.clear();
247 		}
248 		//find the "real" layer and push to array
249 		var fc=polyline.owner;
250 		while(fc && !fc.realLayer){
251 			fc=fc.owner;
252 		}
253 		if(fc){
254 			fc.vectorLayer.lineArray.unshift(polyline);
255 		}
256 		return polyline;
257 	}

360 			}else{

366 				this.totalPointsCounter=0;

532 				default:

598 	this.makeBounds=function(coordinates){
599 		var boundsSouth = 90;
600 		var boundsNorth = -90;
601 		var boundsWest = 180;
602 		var boundsEast = -180;
603 
604 		for(var i=0; i < coordinates.length;i++){
605 			if(coordinates[i] instanceof khtml.maplib.geometry.LatLng){
606 				var lng = coordinates[i].lng();
607                                 var lat = coordinates[i].lat();
608                                 if (lat > boundsNorth) {
609                                         boundsNorth = lat;
610                                 }
611                                 if (lat < boundsSouth) {
612                                         boundsSouth = lat;
613                                 }
614                                 if (lng > boundsEast) {
615                                         boundsEast = lng;
616                                 }
617                                 if (lng < boundsWest) {
618                                         boundsWest = lng;
619                                 }
620 
621 			}else{
622 				var bounds=this.makeBounds(coordinates[i]);
623 				if(bounds.sw().lat() < boundsSouth){
624 					boundsSouth=bounds.sw().lat();
625 				}
626 				if(bounds.sw().lng() < boundsWest){
627 					boundsWest=bounds.sw().lng();
628 				}
629 				if(bounds.ne().lat() > boundsNorth){
630 					boundsNorth=bounds.ne().lat();
631 				}
632 				if(bounds.ne().lng() > boundsEast){
633 					boundsEast=bounds.ne().lng();
634 				}
635 			}
636 		}
637                 //return line bounds
638                 var sw = new khtml.maplib.LatLng(boundsSouth, boundsWest);
639                 var ne = new khtml.maplib.LatLng(boundsNorth, boundsEast);
640                 var b = new khtml.maplib.geometry.Bounds(sw, ne);
641                 return (b);
642 
643 	}
