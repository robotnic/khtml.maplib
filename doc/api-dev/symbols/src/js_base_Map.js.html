283 	this.move = function(evt) {
284 		//khtml.maplib.base.Log.debug('this.move');
285 		
286 		if (evt.preventDefault) {
287 			evt.preventDefault(); // The W3C DOM way
288 		} else {
289 			evt.returnValue = false; // The IE way
290 		}
291 
292 		//	this.mousedownTime=null;
293 		if (evt.touches.length == 1) {
294 			if (this.moveok) {
295 				this.lastMoveX = this.moveX;
296 				this.lastMoveY = this.moveY;
297 				this.lastMoveTime = new Date(evt.timeStamp);
298 				this.moveX = evt.touches[0].pageX / this.faktor / this.sc
299 						+ this.startMoveX;
300 				this.moveY = evt.touches[0].pageY / this.faktor / this.sc
301 						+ this.startMoveY;
302 				if (!this.zoomOutStarted) {
303 					if ((Math.abs(this.moveX - this.oldMoveX) > 5)
304 							|| (Math.abs(this.moveY - this.oldMoveY) > 5)) {
305 						//alert(this.moveX+":"+this.moveY);
306 						window.clearInterval(this.zoomOutInterval);
307 						this.zoomOutSpeed = 0.01;
308 						this.mousedownTime = null;
309 					}
310 					var center = new khtml.maplib.LatLng(this.lat, this.lng);
311 					this.setCenter2(center, this.position.zoom);
312 					this.moveAnimationBlocked = false;
313 				}
314 				if ((Math.abs(this.moveX - this.oldMoveX) > 5)
315 						|| (Math.abs(this.moveY - this.oldMoveY) > 5)) {
316 					//	alert(this.moveX+":"+this.moveX);
317 					//alert(99);
318 					this.mousedownTime = null; //prevents doubleclick if map is moved already
319 				}
320 			} else {
321 				//alert("no move");
322 			}
323 		}
324 
325 		if (evt.touches.length == 2) {
326 			this.mousedownTime = null;
327 			var X1 = evt.touches[0].pageX;
328 			var Y1 = evt.touches[0].pageY;
329 			var X2 = evt.touches[1].pageX;
330 			var Y2 = evt.touches[1].pageY;
331 			var Distance = Math.sqrt(Math.pow((X2 - X1), 2)
332 					+ Math.pow((Y2 - Y1), 2));
333 			var zoomDelta = (Distance / this.startDistance);
334 			var zz = this.startZZ + zoomDelta - 1;
335 			if (zz < 1) {
336 				zz = 1;
337 			}
338 			if (zz > this.tileSource.maxzoom) {
339 				zz = this.tileSource.maxzoom;
340 				zoomDelta = 1;
341 			}
342 			var x = (X1 + X2) / 2;
343 			var y = (Y1 + Y2) / 2;
344 
345 			faktor = Math.pow(2, zz);
346 			var zoomCenterDeltaX = x / faktor - this.width / 2;
347 			var zoomCenterDeltaY = y / faktor - this.height / 2;
348 			var f = Math.pow(2, zoomDelta - 1);
349 			var dx = zoomCenterDeltaX - zoomCenterDeltaX * f;
350 			var dy = zoomCenterDeltaY - zoomCenterDeltaY * f;
351 
352 			this.moveX = (x + dx) / faktor + this.startMoveX;
353 			this.moveY = (y + dy) / faktor + this.startMoveY;
354 
355 			var center = new khtml.maplib.LatLng(this.lat, this.lng);
356 			this.setCenter2(center, zz);
357 		}
358 	}

375 			this.moveok = true;

379 					/*
380 					//khtml.maplib.base.Log.debug('this.end - this.moveAnimationBlocked == false');
381 					var speedX = this.lastMoveX - this.moveX;
382 					var speedY = this.lastMoveY - this.moveY;

387 						*/
388 
389 		var now = new Date(evt.timeStamp);
390 		var timeDelta = now - this.lastMoveTime;
391                                         //speed in pixel per second
392                                         var speedX = (this.lastMoveX - this.moveX) / timeDelta *4;
393                                         var speedY = (this.lastMoveY - this.moveY) / timeDelta *4;
394                                         //console.log(speedX,speedY);
395                                         var maxSpeed = this.wheelSpeedConfig["animateMaxSpeed"];
396                                         if (speedX > maxSpeed)
397                                                 speedX = maxSpeed;
398                                         if (speedY > maxSpeed)
399                                                 speedY = maxSpeed;
400                                         if (speedX < -maxSpeed)
401                                                 speedX = -maxSpeed;
402                                         if (speedY < -maxSpeed)
403                                                 speedY = -maxSpeed;
404 
405                                         var faktor=Math.pow(2,this.zoom());
406                                         if (Math.abs(speedX) > this.wheelSpeedConfig["animateMinSpeed"]/faktor
407                                                         || Math.abs(speedY) > this.wheelSpeedConfig["animateMinSpeed"]/faktor) {
408                                                 this._animateMove(speedX, speedY,faktor);
409                                         } else {
410                                                 //this.renderOverlays();
411                                         }
412                                 }
413 			}
414 		}
