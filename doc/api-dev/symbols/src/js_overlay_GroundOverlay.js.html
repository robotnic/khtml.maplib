 23 /**
 24  * Puts the supplied element at the specified positions.
 25  *
 26  * Marker Object - work in progress
 27  * There should be a possibility to add arbitrary html as marker.
 28  *
 29  * @param {khtml.maplib.LatLng} point
 30  * @param {DOMElement} el
 31  * @param {Object} options
 32  * @class
 33  * @borrows khtml.maplib.overlay.RenderableMixin#_parseOptions as #_parseOptions
 34  * @borrows khtml.maplib.overlay.RenderableMixin#parent as #parent
 35  * @borrows khtml.maplib.overlay.RenderableMixin#map as #map
 36 */
 37 khtml.maplib.overlay.GroundOverlay = function(bounds, el, options) {	
 38 	this.geometry=new Object;
 39 	this.geometry.type="GroundOverlay";
 40 	this.options=options;
 41 	this.type="Feature";
 42 	// ---------------------------------------------------------------
 43 	// Private - should not be used from the outside

 46 	this._createRenderElement = function() {
 47 		var innerDiv = document.createElement("div");
 48 
 49 		// Chain elements
 50 		innerDiv.appendChild(this.el);
 51 		
 52 		// style innerDiv - set defaults
 53 		innerDiv.style.position = "absolute";
 54 		innerDiv.style.top = "0px";
 55 		innerDiv.style.left = "0px";
 56 		innerDiv.style.margin = "0px";
 57 		innerDiv.style.padding = "0px";
 58 		
 59 		// style innerDiv - use options
 60 		if (this.options) {
 61 			if (this.options.dy) {
 62 				innerDiv.style.top = this.options.dy;
 63 			}
 64 			if (this.options.dx) {
 65 				innerDiv.style.left = this.options.dx;
 66 			}
 67 		}
 68 		
 69 		return innerDiv;
 70 	}
 71 	
 72 	/**
 73 	 * calculates element size and set it
 74 	*/
 75 	this._calcImageSize = function() {
 76 		var sw = this.map.latlngToXY(this.bbox.sw());
 77 		var ne = this.map.latlngToXY(this.bbox.ne());
 78 		var nw = this.map.latlngToXY(this.bbox.nw());
 79 		
 80 		// calculates width and height
 81 		dx = ne["x"] - sw["x"];
 82 		dy = sw["y"] - ne["y"];
 83 		
 84 		this.renderElement.style.width = dx;
 85 		this.renderElement.style.height = dy;
 86 
 87 		// Set top-left position
 88 		this.renderElement.style.left = (nw["x"]) + "px";
 89 		this.renderElement.style.top = (nw["y"]) + "px";
 90 		
 91 		var img = this.renderElement.getElementsByTagName("img")[0]
 92 		
 93 		// the image in the div needs to be scaled too
 94 		if(img && img.tagName == "IMG") {
 95 			img.width = dx;
 96 			img.height = dy;
 97 		}
 98 	}

104 	/**
105 	 * Initialize.
106 	 * Applies rotation to "renderElement".
107 	 *
108 	 * @params {Object} options Use options.map and options.parent to assign map/parent. Example: <pre>init({map:myMap, parent: myParent});</pre>
109 	*/
110 	this.init = function(owner) {
111 		this.owner=owner;
112 		if(owner instanceof khtml.maplib.base.Map){
113 			this.map=owner;
114 		}else{
115 			this.map=owner.map;
116 		}
117 		
118 		//this.owner.overlayDiv.appendChild(this.renderElement);
119 //		this._parseOptions(options);
120 		
121 		// debug bounds
122 		// khtml.maplib.base.helpers.showOnMap(this.bounds, this._map);
123 		
124 		// rotate Element
125 		khtml.maplib.base.helpers.rotate(this.renderElement,this.rotation*-1);
126 	}
127 	
128 	/**
129 	 * Render this GroundOverlay.
130 	 *
131 	 * 1. ensure that this Layer has been initialized. (this.init()).
132 	 * 2. execute all pending dom manipulations (this._updateDom()).
133 	 * 3. execute render() on contained layers and features.
134 	*/
135 	this.render = function() {
136 		if (!this.renderElement)
137 			return;
138 		if (!this.point)
139 			return;
140 		if (isNaN(this.point.lat()))
141 			return;
142 		if (isNaN(this.point.lng()))
143 			return;
144 		if(!this.map)
145 			return;
146 			
147 		if(null == this.renderElement.parentNode) {
148 			this.owner.overlayDiv.appendChild(this.renderElement);
149 		}
150 		
151 		// TODO: Bei "this.map().getBounds().intersects(this.bounds)" werden die Koordinaten des nicht
152 		// 		 rotierten Images verwendet. 
153 		//		 Lösung: eine Methode this.boundsRotated() erzeugen, damit mit dieser Fläche verglichen werden kann.
154 		//		 Achtung: Funktioniert die Methode intersects() mit rotierten Rechtecken?
155 
156 			this._calcImageSize();
157 		/*kapier ich nicht
158 		if(this.map.bounds().intersects(this.bbox)) {
159 			this._calcImageSize();
160 		} else {
161 			// remove this image
162 			this.clear();
163 		}
164 		*/
165 	}

167 	/**
168 	 * Set/get bounds of groundoverlay
169 	 */
170 	this.bounds = function(bounds) {
171 		/*
172 		if (bounds) {
173 			//this.bbox = bounds;  //so geht das natuerlich nicht
174 			//console.log("bounding box is read only");
175 		}
176 		*/
177 
178 		if (this.rotation) {
179 			// should return the bounds after rotation
180 			// TODO: calculate rotated Boundingbox; unsure if calculation is correct allready
181 			khtml.maplib.base.Log.log("lng: "+this.bbox.sw().lng()+"; lat:"+this.bbox.sw().lat());
182 			var newSW = this._transformPoint(this.bbox.sw());
183 			khtml.maplib.base.Log.log("lng: "+newSW.lng()+"; lat:"+newSW.lat());
184 			var newSE = this._transformPoint(this.bbox.se());
185 			var newNE = this._transformPoint(this.bbox.ne());
186 			var newNW = this._transformPoint(this.bbox.nw());
187 			
188 			var min_x = Math.min(newSW.lng(), newSE.lng(), newNE.lng(), newNW.lng());			
189 			var min_y = Math.min(newSW.lat(), newSE.lat(), newNE.lat(), newNW.lat());
190 			
191 			var max_x = Math.max(newSW.lng(), newSE.lng(), newNE.lng(), newNW.lng());
192 			var max_y = Math.max(newSW.lat(), newSE.lat(), newNE.lat(), newNW.lat());
193 			
194 			khtml.maplib.base.Log.log("in bounds");
195 						
196 			return new khtml.maplib.geometry.Bounds(new khtml.maplib.LatLng(min_y, min_x), new khtml.maplib.geometry.Point(max_y, max_x));
197 			
198 //			return this.bbox;
199 		} else {
200 			return this.bbox;
201 		}
202 	}
203 	this._transformPoint = function(point) {
204 		var rotCenter = this.bbox.getCenter();
205 		var x0 = rotCenter.lng();
206 		var y0 = rotCenter.lat();
207 		
208 		var theta = this.rotation * Math.PI/180;
209 		
210 		khtml.maplib.base.Log.log("theta: "+theta);
211 		khtml.maplib.base.Log.log("rotation: "+this.rotation);
212 		
213 		var x = point.lng();
214 		var y = point.lat();
215 		
216 		
217 		// source: http://stackoverflow.com/questions/622140/calculate-bounding-box-coordinates-from-a-rotated-rectangle-picture-inside
218 		// transforms the corners of the unrotated boundingbox
219 		var x2 = x0+(x-x0)*Math.cos(theta)+(y-y0)*Math.sin(theta);
220 		var y2 = y0-(x-x0)*Math.sin(theta)+(y-y0)*Math.cos(theta);
221 		
222 		var point= new khtml.maplib.LatLng(y2, x2);
223 		//console.log(y2,x2);
224 		var marker=new khtml.maplib.overlay.SimpleMarker(point,"http://www.din-5008-richtlinien.de/bilder/punkt.gif",{dx:-1,dy:-1});	
225 		map.addOverlay(marker);
226 		return point;
227 	}

229 	/**
230 	 * Set/get position
231 	*/
232 	this.position = function(pos) {
233 		if (pos) {
234 			this.point = pos;
235 		}
236 		return this.point;
237 	}
238 	/**
239 	 * Remove element from DOM.
240 	*/
241 	this.clear = function() {
242 //		this._removeChildRenderElementFromParent(this);
243 		this.renderElement.parentNode.removeChild(this.renderElement);
244 	}
245 	/** Moveable extension */
246 
247 	this.makeMoveable = function() {
248 		khtml.maplib.base.helpers.eventAttach(this.renderElement, "mousedown", this.down, this, false);
249 		khtml.maplib.base.helpers.eventAttach(window, "mousemove", this.move, this, true);
250 		khtml.maplib.base.helpers.eventAttach(window, "mouseup", this.up, this, true);
251 	}
252 	this.moving = false;
253 	this.dx = 0;
254 	this.dy = 0;
255 	this.down = function(evt) {
256 		this.moving = true;
257 		this.dx = this.renderElement.offsetLeft - this._map.pageX(evt)
258 		this.dy = this.renderElement.offsetTop - this._map.pageY(evt)
259 	}
260 	this.move = function(evt) {
261 		if (this.moving) {
262 			var x = this._map.pageX(evt) + this.dx;
263 			var y = this._map.pageY(evt) + this.dy;
264 			//this.renderElement.style.left=x+"px";
265 			//this.renderElement.style.top=y+"px";
266 			this.point = this._map.XYTolatlng(x, y);
267 			this.render();
268 			evt.stopPropagation();
269 		}
270 	}
271 	this.up = function() {
272 		this.moving = false;
273 	}
