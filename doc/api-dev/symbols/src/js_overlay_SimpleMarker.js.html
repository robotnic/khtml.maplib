 23 /**
 24  * Puts the supplied element at the specified positions.
 25  * Has the possibility to add arbitrary html as marker.
 26  *
 27  * DEPRECATED: please use one {khtml.maplib.displayable.Marker}
 28  * {khtml.maplib.displayable.Marker} has the advantage to support {khtml.maplib.overlay.Layer}. (You can put those marker inside layers)
 29  *
 30  * This is intended to ease transition from the previous khtml api to the new, as you can first only
 31  * rename all uses of khtml.maplib.Marker to khtml.maplib.overlay.Marker and then refactor the code to use the
 32  * new {khtml.maplib.displayable.Marker} class.
 33  *
 34  * @deprecated
 35  * @param {khtml.maplib.LatLng} point
 36  * @param {DOMElement} el
 37  * @param {Object} options
 38  * @class
 39 */
 40 khtml.maplib.overlay.SimpleMarker = function(point, el, options) {
 41 	this.nodeType="Marker";
 42 	this.nodeName="Marker";
 43 	this.geometry=new Object();
 44 	this.geometry.coordinates=point;
 45 	this.geometry.type="Point";
 46 	this.type="Marker";
 47 	var div = document.createElement("div");
 48 	div.appendChild(el);
 49 	div.style.position = "absolute";
 50 
 51 	if (options) {
 52 		if (options.dy) {
 53 			if(typeof(options.dy)=="number"){
 54 				div.style.top = options.dy+"px";
 55 			}else{		
 56 				div.style.top = options.dy;
 57 			}
 58 		} else {
 59 			if(typeof(options.dx)=="number"){
 60 				div.style.left = options.dx+"px";
 61 			}else{		
 62 				div.style.left = options.dx;
 63 			}
 64 		}
 65 		if (options.dx) {
 66 			div.style.left = options.dx;
 67 		} else {
 68 			div.style.left = "0px";
 69 		}
 70 	} else {
 71 		div.style.top = "0px";
 72 		div.style.left = "0px";
 73 	}
 74 	var div2 = document.createElement("div");
 75 	div2.style.position = "absolute";
 76 	div2.appendChild(div);
 77 	this.marker = div2;
 78 	this.point = point;
 79 	this.options = options;
 80 	this.el = el
 81 
 82 	this.init = function(owner) {
 83 		this.style=this.marker.style;
 84 		this.marker.owner=this;
 85 		if(owner instanceof khtml.maplib.base.Map){
 86                         this.mapObj=owner;
 87 		}else{
 88 			this.mapObj=owner.map;
 89 		}
 90 		owner.overlayDiv.appendChild(this.marker);
 91 		//this.render();
 92 	}
 93 	this.render = function() {
 94 		if (!this.mapObj)
 95 			return;
 96 		if (!this.marker)
 97 			return;
 98 		if (!this.point)
 99 			return;
100 		if (isNaN(this.point.lat()))
101 			return;
102 		if (isNaN(this.point.lng()))
103 			return;
104 		var xy = this.mapObj.latlngToXY(this.geometry.coordinates);
105 		if (xy["x"] < 0 || xy["y"] < 0) { // <---- flag  ; workaround for overflow:hidden bug
106 			this.marker.style.display = "none";
107 		} else {
108 			this.marker.style.display = "";
109 			this.marker.style.left = xy["x"] + "px";
110 			this.marker.style.top = (xy["y"]) + "px";
111 		}
112 		if (!this.marker.parentNode) {
113 			this.owner.overlayDiv.appendChild(this.marker);
114 		}
115 	}
116 	this.position = function(pos) {
117 		if (pos) {
118 			this.point = pos;
119 			this.render();
120 		}
121 		return this.point;
122 	}
123 
124 	this.clear = function() {
125 		if (this.marker) {
126 			if (this.marker.parentNode) {
127 				try {
128 					this.owner.overlayDiv.removeChild(this.marker);
129 				} catch (e) {
130 				}
131 			}
132 		}
133 	}
134 	/*
135 	 this.moveTo = function (point) {
136 	 this.point = point;
137 	 this.render();
138 	 }
139 	 */
140 
141 	/** Moveable extension */
142 
143 	this.moveable = function(enabled) {
144 		this.enabled=enabled;
145 		if(this.enabled){
146 			this.downevent=khtml.maplib.base.helpers.eventAttach(this.marker, "mousedown", this.down, this, false);
147 			this.moveevent=khtml.maplib.base.helpers.eventAttach(window, "mousemove", this.move, this, true);
148 			this.upevent=khtml.maplib.base.helpers.eventAttach(window, "mouseup", this.up, this, true);
149 		}else{
150 			khtml.maplib.base.helpers.eventRemove(this.downevent);
151 			khtml.maplib.base.helpers.eventRemove(this.moveevent);
152 			khtml.maplib.base.helpers.eventRemove(this.upevent);
153 		}
154 	}
155 	this.moving = false;
156 	this.dx = 0;
157 	this.dy = 0;
158 	this.down = function(evt) {
159 		this.moving = true;
160 		this.dx = this.marker.offsetLeft - this.mapObj.pageX(evt)
161 		this.dy = this.marker.offsetTop - this.mapObj.pageY(evt)
162 	}
163 	this.move = function(evt) {
164 		if (this.moving) {
165 			var x = this.mapObj.pageX(evt) + this.dx;
166 			var y = this.mapObj.pageY(evt) + this.dy;
167 			//this.marker.style.left=x+"px";
168 			//this.marker.style.top=y+"px";
169 			this.point.lat(this.mapObj.XYTolatlng(x, y).lat());
170 			this.point.lng(this.mapObj.XYTolatlng(x, y).lng());
171 			this.render();
172 			evt.stopPropagation();
173 		}
174 	}
175 	this.up = function() {
176 		this.moving = false;
177 	}
178 }
