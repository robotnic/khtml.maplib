 23 /**
 24  * This class understands the OSM-XML Format
 25  *
 26  *
 27  * @class
 28  * @see Example: <a href="../../../examples/parser/osm/start.html">osm with php proxy</a>
 29  * 
 30 */
 31 
 32 
 33 khtml.maplib.parser.Osm=function(){
 34 	this.featureCollection=new khtml.maplib.overlay.FeatureCollection();
 35 	this.nodes=new Array();
 36 	this.ways=new Array();
 37 	this.data=function(){
 38 		return this.featureCollection;
 39 	}
 40 	this.add=function(dom){
 41 		var nodes= dom.getElementsByTagName("node")
 42 		for(var i=0;i<nodes.length;i++){
 43 			var node=nodes[i];
 44 			var id=node.getAttribute("id");
 45 			if(this.nodes[id]){
 46 				//node is already in the array
 47 			}else{
 48 				var tags=node.getElementsByTagName("tag");
 49 				var point=new khtml.maplib.LatLng(node.getAttribute("lat"),node.getAttribute("lon"));
 50 				this.nodes[id]=point;
 51 				var div=document.createElement("div");
 52 				if(tags.length >0){
 53 					var marker=new khtml.maplib.overlay.Marker(point,div);
 54 					marker.properties=new Object;
 55 					for(var t=0;t<tags.length;t++){
 56 						var tag=tags[t];
 57 						var k=tag.getAttribute("k");
 58 						var v=tag.getAttribute("v");
 59 						div.setAttribute("mapcss_"+k,v);
 60 						marker.properties[k]=v;
 61 						//div.style.width="16px";
 62 						//div.style.height="16px";
 63 					}
 64 					div.className="osm_marker";
 65 					
 66 					this.featureCollection.appendChild(marker);
 67 				}
 68 			}
 69 		
 70 		}
 71 
 72 		var ways= dom.getElementsByTagName("way")
 73 		for(var w=0;w<ways.length;w++){
 74 			var way=ways[w];
 75 			var id=way.getAttribute("id");
 76 			if(this.ways[id]){
 77 				//way already in array
 78 			}else{
 79 				var line=new khtml.maplib.geometry.Feature("MultiLineString");
 80 				line.close=true;
 81 				line.id=id;
 82 				var nds=way.getElementsByTagName("nd");
 83 				for(var d=0;d<nds.length;d++){
 84 					var nd=nds[d];
 85 					var ref=nd.getAttribute("ref");
 86 					if(this.nodes[ref]!=undefined){
 87 						line.geometry.coordinates.push(this.nodes[ref]);
 88 					}
 89 				}
 90 				var tags=way.getElementsByTagName("tag");
 91 				for(var t=0;t<tags.length;t++){
 92 					var tag=tags[t];
 93 					var k=tag.getAttribute("k");
 94 					var v=tag.getAttribute("v");
 95 					line.properties[k]=v;
 96 					if(k=="name"){
 97 						line.text.value=v;
 98 						line.text.style.fill="black";
 99 						line.text.style.fontSize="12";
100 						line.text.style.textAnchor="middle";
101 					}
102 				}
103 				this.featureCollection.appendChild(line);
104 				this.ways[id]=line;
105 			}
106 		}
107 
108 	}
109 }
