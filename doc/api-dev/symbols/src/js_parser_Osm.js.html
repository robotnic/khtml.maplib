 23 /**
 24  * This class understands the OSM-XML Format
 25  *
 26  *
 27  * @class
 28 */
 29 
 30 
 31 khtml.maplib.parser.Osm=function(){
 32 	this.featureCollection=new khtml.maplib.overlay.FeatureCollection();
 33 	this.nodes=new Array();
 34 	this.ways=new Array();
 35 	this.data=function(){
 36 		return this.featureCollection;
 37 	}
 38 	this.add=function(dom){
 39 		var nodes= dom.getElementsByTagName("node")
 40 		for(var i=0;i<nodes.length;i++){
 41 			var node=nodes[i];
 42 			var id=node.getAttribute("id");
 43 			if(this.nodes[id]){
 44 				//node is already in the array
 45 			}else{
 46 				var tags=node.getElementsByTagName("tag");
 47 				var point=new khtml.maplib.LatLng(node.getAttribute("lat"),node.getAttribute("lon"));
 48 				this.nodes[id]=point;
 49 				var div=document.createElement("div");
 50 				if(tags.length >0){
 51 					var marker=new khtml.maplib.overlay.Marker(point,div);
 52 					marker.properties=new Object;
 53 					for(var t=0;t<tags.length;t++){
 54 						var tag=tags[t];
 55 						var k=tag.getAttribute("k");
 56 						var v=tag.getAttribute("v");
 57 						div.setAttribute("mapcss_"+k,v);
 58 						marker.properties[k]=v;
 59 						//div.style.width="16px";
 60 						//div.style.height="16px";
 61 					}
 62 					div.className="osm_marker";
 63 					
 64 					this.featureCollection.appendChild(marker);
 65 				}
 66 			}
 67 		
 68 		}
 69 
 70 		var ways= dom.getElementsByTagName("way")
 71 		for(var w=0;w<ways.length;w++){
 72 			var way=ways[w];
 73 			var id=way.getAttribute("id");
 74 			if(this.ways[id]){
 75 				//way already in array
 76 			}else{
 77 				var line=new khtml.maplib.geometry.Feature("Polyline");
 78 				line.close=true;
 79 				line.id=id;
 80 				var nds=way.getElementsByTagName("nd");
 81 				for(var d=0;d<nds.length;d++){
 82 					var nd=nds[d];
 83 					var ref=nd.getAttribute("ref");
 84 					line.geometry.coordinates.push(this.nodes[ref]);
 85 				}
 86 				var tags=way.getElementsByTagName("tag");
 87 				for(var t=0;t<tags.length;t++){
 88 					var tag=tags[t];
 89 					var k=tag.getAttribute("k");
 90 					var v=tag.getAttribute("v");
 91 					line.properties[k]=v;
 92 				}
 93 				this.featureCollection.appendChild(line);
 94 				this.ways[id]=line;
 95 			}
 96 		}
 97 
 98 	}
 99 }
