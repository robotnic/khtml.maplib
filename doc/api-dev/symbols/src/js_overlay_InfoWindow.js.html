 23 /**
 24  * generates a InfoWindowOptions-object<br/>
 25  * options for a infowindow
 26  * @param {string} [content] Content to show in infowindow. Any string, may also contain HTML-tags.
 27  * @param {bool} [disableAutoPan]  not used yet, only for compatibility to Google Markers
 28  * @param {number} [maxWidth] not used yet, only for compatibility to Google Markers
 29  * @param [pixelOffset]  not used yet, only for compatibility to Google Markers
 30  * @param {khtml.maplib.LatLng} [position] LatLng where to place InfoWindow when it's not attached to an anchor
 31  * @param {number} [zIndex]  not used yet, only for compatibility to Google Markers
 32  * @class
 33 */
 34 khtml.maplib.overlay.InfoWindowOptions = function(content, disableAutoPan, maxWidth, pixelOffset, position, zIndex){
 35 	if (content) this.content = content;
 36 	if (position) this.position = position;
 37 }
