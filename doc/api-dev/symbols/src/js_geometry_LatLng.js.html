 23 /**
 24  * Point (2D).
 25  * Implements a copy constructor, use like <pre>new khtml.maplib.LatLng(oldPoint);</pre>
 26  *
 27  * @param {float/khtml.maplib.LatLng/object} lat Latitude - position on y axis (vertical)
 28  * @param {float} lng Longitude - position on x axis (horizontal)
 29  * @class
 30 */
 31 khtml.maplib.geometry.LatLng = function(lat, lng) {
 32         this.callbackFunctions = new Array();
 33 	/**
 34 	Register a callback function or method. "this" will be set to this object in user defined callback function.
 35 	*/
 36         this.addCallbackFunction = function(func) {
 37                 if (typeof (func) == "function") {
 38                         this.callbackFunctions.push(func);
 39                 }
 40         }
 41 	/**
 42 	Internal used to do the callback
 43 	*/
 44         this._executeCallbackFunctions = function() {
 45 		var that=this;
 46                 for ( var i = 0; i < this.callbackFunctions.length; i++) {
 47                         this.callbackFunctions[i].call(that);
 48                 }
 49         }
 50 
 51 
 52 	// Methods
 53 	/**
 54 	set or get latitude
 55 	*/
 56 	this.lat = function(lat) {
 57 		if (typeof(lat) == "number") {
 58 			this.latitude = lat;
 59 			this._executeCallbackFunctions();
 60 		}
 61 		if (typeof(lat) == "string") {
 62 			this.latitude = parseFloat(lat);
 63 			this._executeCallbackFunctions();
 64 		}
 65 		return this.latitude;
 66 	}
 67 	/**
 68 	set or get longitude
 69 	*/
 70 	this.lng = function(lng) {
 71 		if (typeof(lng) == "number") {
 72 			this.longitude = lng;
 73 			this._executeCallbackFunctions();
 74 		}
 75 		if (typeof(lng) == "string") {
 76 			this.longitude = parseFloat(lng);
 77 			this._executeCallbackFunctions();
 78 		}
 79 		return this.longitude;
 80 	}
 81 	
 82 	/**
 83 	 * Position comparison.
 84 	*/
 85 	this.equals = function(other) {   //only same coordinates, not same point
 86 		if(null==other) return false;
 87 		
 88 		if(!(other instanceof khtml.maplib.LatLng)) {
 89 			khtml.maplib.base.Log.warn('Point.equals(): Error - illegal argument!', other);
 90 			return false;
 91 		}
 92 		
 93 		return (other.lat() === this.lat() && other.lng() === this.lng());
 94 	}
 95 
 96 	/**
 97 	// Constructor
 98 	*/
 99 	if(lat instanceof khtml.maplib.geometry.LatLng) {	// ewi: changed khtml.maplib.LatLng to khtml.maplib.geometry.LatLng
100 		// Copy constructor
101 		this.lat(lat.lat());
102 		this.lng(lat.lng());
103 	} else if (typeof(lat) == "object") {
104 		// Input as object with named values {lat: 40.1, lng: 5.676}
105 		if(lat.lat) {
106 			this.lat(lat.lat);
107 		}
108 		if(lat.lng) {
109 			this.lng(lat.lng);
110 		}
111 	} else {
112 		// Default
113 		this.lat(lat);
114 		this.lng(lng);
115 	}
116 
117 
118 }
