291 function translateToSVG(data){
292         for(var i=0;i<data.length;i++){
293                 for(s in data[i].rules){
294                         switch(s){
295                                 case "color":data[i].rules["stroke"]=data[i].rules[s];
296                                         break;
297                                 case "fill-color":data[i].rules["fill"]=data[i].rules[s];
298                                         break;
299                                 case "dashes":data[i].rules["stroke-dasharray"]=data[i].rules[s];
300                                         break;
301                                 case "width":data[i].rules["stroke-width"]=data[i].rules[s];
302                                         break;
303                         }
304                 }
305         }
306         return(data);
307 }

317 function removeComments(str) {
318     str = ('__' + str + '__').split('');
319     var mode = {
320         singleQuote: false,
321         doubleQuote: false,
322         regex: false,
323         blockComment: false,
324         lineComment: false,
325         condComp: false
326     };
327     for (var i = 0, l = str.length; i < l; i++) {
328 
329         if (mode.regex) {
330             if (str[i] === '/' && str[i-1] !== '\\') {
331                 mode.regex = false;
332             }
333             continue;
334         }
335 
336         if (mode.singleQuote) {
337             if (str[i] === "'" && str[i-1] !== '\\') {
338                 mode.singleQuote = false;
339             }
340             continue;
341         }
342 
343         if (mode.doubleQuote) {
344             if (str[i] === '"' && str[i-1] !== '\\') {
345                 mode.doubleQuote = false;
346             }
347             continue;
348         }
349 
350         if (mode.blockComment) {
351             if (str[i] === '*' && str[i+1] === '/') {
352                 str[i+1] = '';
353                 mode.blockComment = false;
354             }
355             str[i] = '';
356             continue;
357         }
358 
359         if (mode.lineComment) {
360             if (str[i+1] === '\n' || str[i+1] === '\r') {
361                 mode.lineComment = false;
362             }
363             str[i] = '';
364             continue;
365         }
366 
367         if (mode.condComp) {
368             if (str[i-2] === '@' && str[i-1] === '*' && str[i] === '/') {
369                 mode.condComp = false;
370             }
371             continue;
372         }
373 
374         mode.doubleQuote = str[i] === '"';
375         mode.singleQuote = str[i] === "'";
376 
377         if (str[i] === '/') {
378 
379             if (str[i+1] === '*' && str[i+2] === '@') {
380                 mode.condComp = true;
381                 continue;
382             }
383             if (str[i+1] === '*') {
384                 str[i] = '';
385                 mode.blockComment = true;
386                 continue;
387             }
388             if (str[i+1] === '/') {
389                 str[i] = '';
390                 mode.lineComment = true;
391                 continue;
392             }
393             mode.regex = true;
394 
395         }
396 
397     }
398     return str.join('').slice(2, -2);
399 }

402 function trim (zeichenkette) {
403   // Erst führende, dann Abschließende Whitespaces entfernen
404   // und das Ergebnis dieser Operationen zurückliefern
405   return zeichenkette.replace (/^\s+/, '').replace (/\s+$/, '');
406 }
