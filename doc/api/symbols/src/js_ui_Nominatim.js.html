1 /**
  2 Search with nominatim
  3 */
  4 
  5 khtml.maplib.ui.Nominatim=function(){
  6 	this.width=300;
  7 	this.height=200;
  8 	this.init=function(map){
  9 		this.map=map;
 10 		this.div=this.createWindow();
 11 	}
 12 	this.render=function(){
 13 	//nothing to do
 14 	}
 15 	var that=this;
 16 	this.keyup=function(evt){
 17 		var searchText=that.input.value;
 18 		if(searchText.length <=2){
 19 			while(that.div.firstChild){
 20 				that.div.removeChild(that.div.firstChild);
 21 			}
 22 			return;
 23 		}
 24 		var url="http://nominatim.openstreetmap.org/search?q="+searchText+"&format=json&polygon=1&addressdetails=1&json_callback=searchResult";
 25 		khtml.maplib.util.scriptTagProxy(url,that.showResult);
 26 	}
 27 	this.showResult=function(r){
 28 		that.div.style.display="";
 29 		while(that.div.firstChild){
 30 			that.div.removeChild(that.div.firstChild);
 31 		}
 32 		for(var i=0; i< r.length;i++){
 33 			var found=r[i];
 34 			var div=document.createElement("div");
 35 			var tn=document.createTextNode(found.display_name);
 36 			div.appendChild(tn);
 37 			div.style.fontSize="12px";
 38 			div.style.width="1200px";
 39 			div.style.cursor="pointer";
 40 			that.div.appendChild(div);
 41 			khtml.maplib.base.helpers.eventAttach(div,"mouseover",that.highlight,div,false);
 42 			khtml.maplib.base.helpers.eventAttach(div,"mouseout",that.lowlight,div,false);
 43 			khtml.maplib.base.helpers.eventAttach(div,"mousedown",that.select,found,false);
 44 		}
 45 	}
 46 	this.hideResult=function(){
 47 		var tempFunc=function(){
 48 		that.div.style.display="none";
 49 		}
 50 		this.timeout=setTimeout(tempFunc,1000);
 51 	}
 52 	this.unhideResult=function(){
 53 		that.div.style.display="";
 54 		clearTimeout(this.timeout);
 55 	}
 56 
 57 	this.highlight=function(){
 58 		this.style.backgroundColor="yellow";
 59 	}
 60 	this.lowlight=function(){
 61 		this.style.backgroundColor="";
 62 	}
 63 	this.select=function(){
 64 		var sw=new khtml.maplib.LatLng(this.boundingbox[0],this.boundingbox[2]);
 65 		var ne=new khtml.maplib.LatLng(this.boundingbox[1],this.boundingbox[3]);
 66 		var bounds=new khtml.maplib.geometry.Bounds(sw,ne);
 67 		that.map.bounds(bounds);
 68 		that.div.style.display="none";
 69 
 70 		that.showPoly(this.polygonpoints);
 71 	}
 72 	this.showPoly=function(coordinates){
 73 		var poly=new khtml.maplib.geometry.Feature({type:"LineString",coordinates:coordinates});
 74 		poly.style.fill="none";
 75 		poly.style.stroke="red";
 76 		poly.style.strokeWidth="4";
 77 		this.map.featureCollection.appendChild(poly);
 78 	}
 79 	this.createWindow=function(){
 80 		var win=document.createElement("div");
 81 
 82 		this.input=document.createElement("input");
 83 		this.input.style.width="300px";
 84 		this.input.style.height="20px";
 85 		this.input.style.border="1px solid black";
 86 		win.appendChild(this.input);
 87 		khtml.maplib.base.helpers.eventAttach(this.input,"keyup",this.keyup,this,false);
 88 
 89 
 90 		var div=document.createElement("div");
 91 		win.style.position="absolute";
 92 		div.style.width=this.width+"px";
 93 		//div.style.height=this.height+"px";
 94 		var x=this.map.size.left;
 95 		var y=this.map.size.top;
 96 		win.style.top=y+"px";
 97 		win.style.left=x+"px";
 98 		div.style.border="1px solid lightgrey";
 99 		div.style.backgroundColor="white";
100 		div.style.opacity=0.8;
101 		div.style.display="none";
102 		div.style.overflow="hidden";
103 
104 		div.style.borderBottomRightRadius="10px";
105 		div.style.borderBottomLeftRadius="10px";
106 		div.style.boxShadow="1px 1px 15px #ccc";
107 		khtml.maplib.base.helpers.eventAttach(win,"mouseout",this.hideResult,this,false);
108 		khtml.maplib.base.helpers.eventAttach(win,"mouseover",this.unhideResult,this,false);
109 
110 		win.appendChild(div);
111 		this.map.mapParent.parentNode.appendChild(win);
112 		return div;	
113 	}
114 
115 }
