 24 //zoombar and scrollhandle

 50 /**
 51  * Zoombar
 52  *
 53  * @class
 54 */
 55 khtml.maplib.ui.Zoombar = function() {
 56 	// -------------------------------
 57 	// Private Methods.
 58 	// Intended to be accessed internaly only
 59 	// -------------------------------
 60 	
 61 	/**
 62 	 * Cancels the event if it is cancelable, 
 63 	 * without stopping further propagation of the event.
 64 	*/
 65 	this._cancelEvent = function(evt) {
 66 		evt.cancelBubble = true;
 67 		if (evt.stopPropagation)
 68 			evt.stopPropagation();
 69 	}
 70 	/**
 71 	 * Prevents further propagation of the current event.
 72 	*/
 73 	this._stopEventPropagation = function(evt) {
 74 		if (evt.preventDefault) {
 75 			evt.preventDefault(); // The W3C DOM way
 76 		} else {
 77 			evt.returnValue = false; // The IE way
 78 		}		
 79 	}
 80 	/**
 81 	 * Calculate Z if Y is given. (also use in testcase)
 82 	*/
 83 	this._calcZFromY = function(y) {
 84 		return 22 - y / 10;
 85 	}
 86 	/**
 87 	 * Zoom
 88 	*/
 89 	this._zoom = function(evt) {
 90 		if ((evt.type == "touchstart") || (evt.type == "touchmove"))
 91 			var y = this.themap.pageY(evt.touches[0]);
 92 		else
 93 			var y = this.themap.pageY(evt);
 94 		this.themap.zoom(this._calcZFromY(y));
 95 	}
 96 	
 97 	/**
 98 	 * Mousedown or Touchstart
 99 	*/
100 	this._down = function(evt) {
101 		this._cancelEvent(evt);
102 		
103 		this.moving = true;
104 		
105 		this._zoom(evt);
106 		
107 		this._stopEventPropagation(evt);
108 	}
109 	/**
110 	 * Called while scrolling and Touchmove
111 	*/
112 	this._move = function(evt) {
113 		this._cancelEvent(evt);
114 		
115 		if (this.moving) {
116 			this._zoom(evt);
117 			this.render();
118 			this._stopEventPropagation(evt);
119 		}
120 	}
121 	/**
122 	 * Mouseup or Touchend
123 	*/
124 	this._up = function(evt) {
125 		this._cancelEvent(evt);
126 		
127 		this.moving = false;
128 		
129 		this._stopEventPropagation(evt);
130 	}
131 	
132 	// -------------------------------
133 	// Public Methods.
134 	// Intended to be called from the outside.
135 	// -------------------------------
136 	
137 	/**
138 	 * called by maplib once
139 	 */
140 	this.init = function(themap) {
141 		this.themap = themap;
142 		var els = themap.mapParent.getElementsByTagName("*");
143 		for ( var i = 0; i < els.length; i++) {
144 			var el = els.item(i);
145 			if (el.className == "scrollhandle") {
146 				this.scrollhandle = el;
147 			}
148 			if (el.className == "zoombar") {
149 				this.zoombar = el;
150 			}
151 		}
152 		if (navigator.userAgent.indexOf("MSIE") != -1) {		// IE does not support window-Eventhandler
153 			var w = this.mapObj.mapParent;
154 		} else {
155 			var w = window;
156 		}
157 		khtml.maplib.base.helpers.eventAttach(this.zoombar, "mousedown", this.down, this, false);
158 		khtml.maplib.base.helpers.eventAttach(this.zoombar, "mousemove", this.move, this, false);
159 		khtml.maplib.base.helpers.eventAttach(w, "mouseup", this.up, this, false);
160 		
161 		khtml.maplib.base.helpers.eventAttach(this.zoombar, "touchstart", this._down, this, false);
162 		khtml.maplib.base.helpers.eventAttach(this.zoombar, "touchmove", this._move, this, false);
163 		khtml.maplib.base.helpers.eventAttach(this.zoombar, "touchend", this._up, this, false);
164 	}
165 	/**
166 	 * called by maplib on every map change
167 	 */
168 	this.render = function() {
169 		var top = (22 - this.themap.zoom()) * 10;
170 		this.scrollhandle.style.marginTop = top + "px";
171 	}
172 }
