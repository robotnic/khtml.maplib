 12 	this.type="FeatureCollection";
 13 	this.style=new Object;
 14 	this.className=new Object;
 15 	if(parentNode){
 16 		this.parentNode=parentNode;
 17 		if(!parentNode.documentElement){
 18 			this.documentElement=parentNode;
 19 		}else{
 20 			this.documentElement=this.parentNode.documentElement;
 21 		}
 22 		this.map=this.parentNode.map;
 23 		//this.backend=this.parentNode.backend;
 24 		if(this.parentNode.context)this.context=this.parentNode.context;
 25 		if(this.documentElement.backend=="svg"){
 26 			this.element=document.createElementNS("http://www.w3.org/2000/svg","g");
 27 			if(this.parentNode.element){
 28 				this.parentNode.element.appendChild(this.element);
 29 			}
 30 		}

 32 	if(!this.features){
 33 		this.features=new Array();
 34 	}
 35 	this.childNodes=this.features;
 36 	if(this.documentElement && this.documentElement.backend=="svg"){
 37 		if(feature.style){
 38 			for(var s in feature.style){
 39 				this.style[s]=feature.style[s];
 40 			}

138 	this.insertBefore=function(newFeature,refFeature){
139 		var exists=false;
140 		var numDelete=0;
141 		if(!refFeature){
142 			if(this.features[0]){
143 				refFeature=this.features[0];
144 			}else{
145 				this.appendChild(newFeature);
146 				return false;
147 			}
148 		}
149 		if(newFeature==refFeature)return newFeature; //already at correct position
150 		if(!newFeature.element && !newFeature.marker){
151 			newFeature=new khtml.maplib.geometry.Feature(newFeature);
152 		}else{
153 			exists=true;
154 			if(newFeature.parentNode){
155 			for(var i=0;i<newFeature.parentNode.features.length;i++){
156 				if(newFeature.parentNode.features[i]==newFeature){
157 					newFeature.parentNode.features.splice(i,1);
158 					break;
159 				}
160 			}
161 			}
162 		}
163 		for(var i=0;i<this.features.length;i++){
164 			if(this.features[i]==refFeature){
165 				this.features.splice(i,0,newFeature);
166 				if(newFeature.element){	
167 					this.element.insertBefore(newFeature.element,refFeature.element);
168 				}
169 				break;
170 			}
171 		}
172 		this.documentElement.recurseLinksDirty=true;
173 		newFeature.parentNode=this;
174 		if(!exists){
175 			newFeature.init(this);
176 		}
177 		newFeature.render();
178 		this.makeBounds();
179 
180 		return newFeature;
181 	}

194 				}
195 				if(this.features[i].following){
196 					this.features[i].following.preceding=this.features[i].preceding;
197 				}
198 				/*

200 				this.removeChild(this.features[i]);
201 				}catch(e){}
202 				*/
203 
204 				try{
205 				this.element.removeChild(this.features[i].element);
206 				}catch(e){
207 					//console.log(e.message);
208 				}
209 				this.features.splice(i,1);
210 				break;
211 			}
212 		}
213 		if(this.documentElement.backend=="canvas"){
214 			this.documentElement.render(true);
215 		}
216 		
217 	}
