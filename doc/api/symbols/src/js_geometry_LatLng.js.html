 23 /**
 24  * Point (2D).
 25  * Implements a copy constructor, use like <pre>new khtml.maplib.LatLng(oldPoint);</pre>
 26  *
 27  * @param {float/khtml.maplib.LatLng/object} lat Latitude - position on y axis (vertical)
 28  * @param {float} lng Longitude - position on x axis (horizontal)
 29  * @class
 30 */
 31 khtml.maplib.geometry.LatLng = function(lat, lng) {
 32         this.callbackFunctions = new Array();
 33         this.addCallbackFunction = function(func) {
 34                 if (typeof (func) == "function") {
 35                         this.callbackFunctions.push(func);
 36                 }
 37         }
 38         this.executeCallbackFunctions = function() {
 39 		var that=this;
 40                 for ( var i = 0; i < this.callbackFunctions.length; i++) {
 41                         this.callbackFunctions[i].call(that);
 42                 }
 43         }
 44 
 45 
 46 	// Methods
 47 	this.lat = function(lat) {
 48 		if (typeof(lat) == "number") {
 49 			this.latitude = lat;
 50 			this.executeCallbackFunctions();
 51 		}
 52 		if (typeof(lat) == "string") {
 53 			this.latitude = parseFloat(lat);
 54 			this.executeCallbackFunctions();
 55 		}
 56 		return this.latitude;
 57 	}
 58 	this.lng = function(lng) {
 59 		if (typeof(lng) == "number") {
 60 			this.longitude = lng;
 61 			this.executeCallbackFunctions();
 62 		}
 63 		if (typeof(lng) == "string") {
 64 			this.longitude = parseFloat(lng);
 65 			this.executeCallbackFunctions();
 66 		}
 67 		return this.longitude;
 68 	}
 69 	
 70 	/**
 71 	 * Position comparison.
 72 	*/
 73 	this.equals = function(other) {   //only same coordinates, not same point
 74 		if(null==other) return false;
 75 		
 76 		if(!(other instanceof khtml.maplib.LatLng)) {
 77 			khtml.maplib.base.Log.warn('Point.equals(): Error - illegal argument!', other);
 78 			return false;
 79 		}
 80 		
 81 		return (other.lat() === this.lat() && other.lng() === this.lng());
 82 	}
 83 
 84 	// ---------------------
 85 	// Constructor
 86 	// ---------------------
 87 	if(lat instanceof khtml.maplib.LatLng) {
 88 		// Copy constructor
 89 		this.lat(lat.lat());
 90 		this.lng(lat.lng());
 91 	} else if (typeof(lat) == "object") {
 92 		// Input as object with named values {lat: 40.1, lng: 5.676}
 93 		if(lat.lat) {
 94 			this.lat(lat.lat);
 95 		}
 96 		if(lat.lng) {
 97 			this.lng(lat.lng);
 98 		}
 99 	} else {
100 		// Default
101 		this.lat(lat);
102 		this.lng(lng);
103 	}
104 
105 
106 }
