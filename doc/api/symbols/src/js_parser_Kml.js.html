 23 /**
 24  * class understands KML and manages the layers related to the KML passed
 25  * is like a layer in the view of the map but manages multiple layers of different types
 26  * can be added to map as overlay
 27  *
 28  * @class
 29  * @see Example: <a href="../../../examples/parser/kml/googlebuildings.html">kml from google tutorial</a>
 30 */
 31 khtml.maplib.parser.Kml = function(KML) {
 32 	// initialize variables
 33 	this.parsed = null;
 34 	this.styles=new Array();
 35 	
 36 	// Methods
 37 	/**
 38 	 * Parse KML!
 39 	 *
 40 	 * @param {boolean} force	parse dom again - even if it has been parsed before.
 41 	 * @returns {khtml.maplib.overlay.Layer}
 42 	*/
 43 	this.parse = function(force) {
 44 		//khtml.maplib.base.Log.log('Kml.parse(): begin');
 45 		if(force || null==this.parsed) {
 46 			if(!this.dom) {
 47 				//khtml.maplib.base.Log.warn('Kml.parse(): Aborted. dom is empty! use kml.setDom(data)');
 48 				return false;
 49 			}
 50 
 51 			if(this.dom.hasChildNodes) {
 52 				this.parsed = this.parseNode(this.dom.documentElement);
 53 			}
 54 		}
 55 		
 56 		//khtml.maplib.base.Log.log('Kml.parse(): end');
 57 		
 58 		return this.parsed;
 59 	}
 60 
 61 	/**
 62 	 * Parse a single node.
 63 	 * Entry point for parsing.
 64 	 *
 65 	 * @param {DOMElement} node
 66 	 * @returns {khtml.maplib.overlays.Layer} or null
 67 	*/
 68 	this.parseNode = function(node) {		
 69 		if(!node) return false;
 70 		
 71 		var layer = null;
 72 		var newLayer = null;
 73 		
 74 		
 75 		switch(node.nodeName) {
 76 			case "xml":
 77 			case "#document":  //??????
 78 				if(khtml.maplib.base.helpers.isIE() && node.childNodes.length > 1) return this.parseNode(node.childNodes[1]);
 79 				return this.parseNode(node.childNodes[0]);
 80 				break;
 81 			case "kml":
 82 				/*
 83 				if(node.childNodes.length > 1) {
 84 					khtml.maplib.base.Log.warn(
 85 						"Some elements inside the <kml>-Tag were ignored! Only the first tag was parsed!"
 86 					);
 87 				}
 88 				return this.parseNode(node.childNodes[0]);
 89 				*/
 90 				var fc=this.parseNode(node.getElementsByTagName("*").item(0));
 91 				return fc;
 92 				/*
 93 				for(var j = 0; j < node.childNodes.length; j++) {
 94 					var layer = this.parseNode(node.childNodes[j]);
 95 					if(layer != null) return layer;
 96 				}
 97 				*/
 98 				break;
 99 			case "Document":
100 				//newLayer = this.parseDocument(node);
101 				layer = this.parseDocument(node);
102 				return layer;
103 			case "Folder":
104 				if ("Folder" == node.nodeName) {
105 					newLayer = this.parseFolder(node);
106 				}
107 
108 				if(null==layer) {
109 					layer = newLayer;
110 				} else {
111 					layer.appendChild(newLayer);
112 				}
113 				break;
114 			case "Placemark":
115 				// prepare layer
116 				//if(null==layer) layer = new khtml.maplib.overlay.Layer();
117 				// add feature
118 				layer.appendChild(this.parsePlacemark(node));
119 				break;
120 			case "GroundOverlay":
121 				// prepare layer
122 				if(null==layer) layer = new khtml.maplib.overlay.FeatureCollection();
123 				// add feature
124 				layer.appendChild(this.parseGroundOverlay(node));
125 				break;
126 			case "Region":
127 			case "ExtendedData":
128 
129 			default:
130 				//khtml.maplib.base.Log.log("ignored node (parseNode): " +  + node.nodeName);
131 				break;
132 		}
133 		
134 		return layer;
135 	}

137 	/**
138 	 * Parse "Folder" tags in the first level of the supplied node.
139 	 * This function dosen't support to parse all tags because folder
140 	 * structure is allways hierarchical.
141 	 *
142 	 *
143 	 * <pre>A Folder is used to arrange other Features hierarchically (Folders, Placemarks, NetworkLinks, or Overlays).
144 	 * A Feature is visible only if it and all its ancestors are visible.</pre>
145 	 *
146 	 * @see http://code.google.com/intl/de/apis/kml/documentation/kmlreference.html#folder 
147 	 * @param {DOMElement} node
148 	*/
149 	this.parseFolder = function(node) {
150 		//khtml.maplib.base.Log.log('parse Folder!');
151 		var folder = new khtml.maplib.overlay.FeatureCollection();
152 
153 		for(var j = 0; j < node.childNodes.length; j++) {	
154 			switch(node.childNodes[j].nodeName) {
155 				case "name":
156 					folder.properties.name=node.childNodes[j].childNodes[0].nodeValue;
157 					break;
158 				case "open":
159 					// Specifies if the folder is open/closed in Google Earths list view.
160 					// 1=expanded / 0=collapsed
161 					//folder.attribute("open", node.childNodes[j].childNodes[0].nodeValue);
162 					break;
163 				case "visibility":
164 					// Specifies whether the feature is drawn in the 3D viewer when it is initially loaded.
165 					// In order for a feature to be visible, the <visibility> tag of all its ancestors
166 					// must also be set to 1.
167 					folder.visible(parseInt(node.childNodes[j].childNodes[0].nodeValue)==1);
168 					break;
169 				case "description":
170 					folder.properties.description=node.childNodes[j].childNodes[0].nodeValue;
171 					break;
172 				case "Folder":
173 					folder.appendChild(this.parseFolder(node.childNodes[j]));
174 					break;
175 				case "Placemark":
176 					// add feature
177 					folder.appendChild(this.parsePlacemark(node.childNodes[j]));
178 					break;
179 				case "GroundOverlay":
180 					folder.appendChild(this.parseGroundOverlay(node.childNodes[j]));
181 					break;
182 
183 				default:
184 					//khtml.maplib.base.Log.log("ignored node (parseFolder): " + node.childNodes[j].nodeName);
185 			}
186 		}
187 
188 		//khtml.maplib.base.Log.log('Folder:', folder, node);
189 		
190 		return folder;
191 	}

193 	/**
194 	 * Parse a "Document" tag.
195 	 *
196 	 * <pre>A Document is a container for features and styles.
197 	 * This element is required if your KML file uses shared styles. 
198 	 * Do not put shared styles within a Folder.</pre>
199 	 *
200 	 * @see http://code.google.com/intl/de/apis/kml/documentation/kmlreference.html#document 
201 	 * @param {DOMElement} node
202 	*/
203 	this.parseDocument = function(node) {
204 		var doc = new khtml.maplib.overlay.FeatureCollection();
205 
206 		
207 		for(var j = 0; j < node.childNodes.length; j++) {	
208 			if(node.childNodes[j].nodeType!=1)continue; //white spaces
209 			switch(node.childNodes[j].nodeName) {
210 				case "name":
211 					doc.name=node.childNodes[j].childNodes[0].nodeValue;
212 					break;
213 				case "open":
214 					// Specifies if the folder is open/closed in Google Earths list view.
215 					// 1=expanded / 0=collapsed
216 					doc.attribute("open", node.childNodes[j].childNodes[0].nodeValue);
217 					break;
218 				case "visibility":
219 					// Specifies whether the feature is drawn in the 3D viewer when it is initially loaded.
220 					// In order for a feature to be visible, the <visibility> tag of all its ancestors
221 					// must also be set to 1.
222 					doc.visible(parseInt(node.childNodes[j].childNodes[0].nodeValue)==1);
223 					break;
224 				case "description":
225 					// Text in Popup Baloon
226 					doc.properties.description=node.childNodes[j].childNodes[0].nodeValue;
227 					break;
228 				case "Style":
229 					var style = this.parseStyle(node.childNodes[j]);
230 					break;
231 				case "Folder":
232 					doc.appendChild(this.parseFolder(node.childNodes[j]));
233 					break;
234 				case "GroundOverlay":
235 					doc.addContent(this.parseGroundOverlay(node.childNodes[j]), true);
236 					break;
237 				case "Placemark":
238 					var feature=this.parsePlacemark(node.childNodes[j]);
239 					if(feature.type=="Feature" || feature.type=="Marker"){
240 						doc.appendChild(feature);
241 					}else{
242 					}
243 					break;
244 				default:
245 					//khtml.maplib.base.Log.log("ignored node (parseDocument): " + node.childNodes[j].nodeName);
246 			}
247 		}
248 		
249 		//khtml.maplib.base.Log.log('Document:', doc);
250 		
251 		return doc;
252 	}

254 	/**
255 	 * Parse a "Style" tag.
256 	 *
257 	 * @see http://code.google.com/intl/de/apis/kml/documentation/kmlreference.html#style
258 	 * @param {DOMElement} node XML Node
259 	*/
260 	this.parseStyle = function(node) {
261 		if(!node) return false;
262 		var id=node.getAttribute("id");
263 		if(!this.styles[id]){
264 			this.styles[id]=new Array();
265 		}
266 		for(var i=0;i<node.childNodes.length;i++){
267 			var styleNode=node.childNodes[i];
268 			if(styleNode.nodeType!=1)continue;
269 			var style=new Object;
270 			style.fill="none";	
271 			style.stroke="rgba(255,0,0,0.48828125)";
272 			style.strokeWidth=3;	
273 			this.styles[id][style.tagName]=style;
274 			for(var p=0; p< styleNode.childNodes.length;p++){
275 				if(styleNode.childNodes[p].nodeType!=1)continue;
276 				var tagName=styleNode.childNodes[p].tagName;
277 				if(!styleNode.childNodes[p].firstChild)continue; //empty element
278 				switch(tagName){
279 					case "color":
280 						if(styleNode.nodeName=="LineStyle"){
281 							 style.stroke=hex2rgba(styleNode.childNodes[p].firstChild.nodeValue);
282 						}
283 						if(styleNode.nodeName=="PolyStyle") style.fill=hex2rgba(styleNode.childNodes[p].firstChild.nodeValue);
284 						break;
285 					case "width":
286 						if(styleNode.nodeName=="LineStyle") style.strokeWidth=styleNode.childNodes[p].firstChild.nodeValue;
287 						if(styleNode.nodeName=="PolyStyle") style.strokeWidth=styleNode.childNodes[p].firstChild.nodeValue;
288 						break;
289 					default:
290 						console.log("not implemented",tagName,styleNode[p]);
291 				}
292 			}
293 			this.styles[id][styleNode.tagName]=style;
294 			
295 		}
296 		
297 		//khtml.maplib.base.Log.warn('parseStyle(): Warning - not implemented!');
298 		return false;
299 	}	
300 	/**
301 	 * Parse a "GroundOverlay" tag.
302 	 *
303 	 * @example
304 	 <pre>
305 	<GroundOverlay>
306 	   <name>GroundOverlay.kml</name>
307 	   <color>7fffffff</color>
308 	   <drawOrder>1</drawOrder>
309 	   <Icon>
310 	      <href>http://www.google.com/intl/en/images/logo.gif</href>
311 	      <refreshMode>onInterval</refreshMode>
312 	      <refreshInterval>86400</refreshInterval>
313 	      <viewBoundScale>0.75</viewBoundScale>
314 	   </Icon>
315 	   <LatLonBox>
316 	      <north>37.83234</north>
317 	      <south>37.832122</south>
318 	      <east>-122.373033</east>
319 	      <west>-122.373724</west>
320 	      <rotation>45</rotation>
321 	   </LatLonBox>
322 	</GroundOverlay>
323 	</pre>
324 	 *
325 	 * Note: rotation is in degree (+-180) around object center.
326 	 *
327 	 * @see http://code.google.com/intl/de/apis/kml/documentation/kmlreference.html#groundoverlay
328 	 * @param {DOMElement} node XML Node of type "GroundOverlay"
329 	 * @returns false if node is not a groundoverlay. Otherwise a new {khtml.maplib.overlay.Image} layer.
330 	*/
331 	this.parseGroundOverlay = function(node) {
332 		if(!node) return false;
333 		if(!node.nodeName=="GroundOverlay") return false;
334 		
335 		// <Icon>
336 		var myIcon = this.parseIcon(node);
337 		
338 		// <LatLonBox>
339 		var latlonbox = node.getElementsByTagName("LatLonBox")[0];
340 		
341 		var north = latlonbox.getElementsByTagName("north")[0].childNodes[0].nodeValue;
342 		var east = latlonbox.getElementsByTagName("east")[0].childNodes[0].nodeValue;
343 		
344 		var south = latlonbox.getElementsByTagName("south")[0].childNodes[0].nodeValue;
345 		var west = latlonbox.getElementsByTagName("west")[0].childNodes[0].nodeValue;
346 		
347 		// <LatLonBox><rotation>..
348 		var options = new Object();
349 		if(latlonbox.getElementsByTagName("rotation")[0] != undefined) {
350 			// fetch rotation
351 			var myRotation = latlonbox.getElementsByTagName("rotation")[0].childNodes[0].nodeValue;
352 			options.rotation = parseFloat(myRotation);
353 		}
354 		
355 		// Define Boundingbox
356 		var myBB = new khtml.maplib.geometry.Bounds(
357 			new khtml.maplib.LatLng(south, west),
358 			new khtml.maplib.LatLng(north, east)
359 		);
360 		
361 		// Create new image layer
362 		var myLayer = new khtml.maplib.overlay.GroundOverlay(myBB, myIcon, options);
363 
364 		// Create feature
365 		/*
366 		var feature = new khtml.maplib.overlay.Feature();
367 		feature.marker(myLayer);
368 		feature.geometry(myBB);
369 		*/
370 		
371 		//khtml.maplib.base.Log.log('GroundOverlay:', feature);
372 
373 		return myLayer;
374 	}

376 	/**
377 	 * Parse a "Placemark" tag.
378 	 *
379 	 * @param {DOMElement} node
380 	*/
381 	this.parsePlacemark = function(node) {
382 		if(!node) return false;
383 		var pm = node;
384 		var data = {};
385 		data.visibility = true; // default
386 		//khtml.maplib.base.Log.log('node',node);
387 		
388 		for(var j = 0; j < pm.childNodes.length; j++) {
389 			switch(pm.childNodes[j].nodeName) {
390 				case "name":
391 					if(pm.childNodes[j].childNodes.length > 0) {
392 						if(typeof(pm.childNodes[j].childNodes[0].nodeValue)!='undefined'){
393 							data.name = pm.childNodes[j].childNodes[0].nodeValue;
394 						}
395 					}
396 					break;
397 				case "description":
398 					if(pm.childNodes[j].childNodes.length > 0) {
399 						if(typeof(pm.childNodes[j].childNodes[0].nodeValue)!='undefined'){
400 							data.description = pm.childNodes[j].childNodes[0].nodeValue;
401 						}
402 					}
403 					break;
404 				case "Point":
405 					data = this.parsePoint(pm.childNodes[j]);
406 					break;
407 				case "LineString":
408 					data = this.parseLineString(pm.childNodes[j]);
409 					break;
410 				case "Polygon":
411 					data = this.parsePolygon(pm.childNodes[j]);
412 					break;
413 				case "gx:drawOrder":
414 					data.gxdraworder = pm.childNodes[j];
415 					break;
416 				case "visibility":
417 					// Specifies whether the feature is drawn in the 3D viewer when it is initially loaded.
418 					// In order for a feature to be visible, the <visibility> tag of all its ancestors
419 					// must also be set to 1.
420 					data.visibility = parseInt(pm.childNodes[j].childNodes[0].nodeValue)==1;
421 					break;
422 			}
423 		
424 			//khtml.maplib.base.Log.log('data',data);
425 			var styleUrls=pm.getElementsByTagName("styleUrl");
426 			if(styleUrls.length>0){
427 				if(styleUrls[0].firstChild){
428 					var stylename=styleUrls[0].firstChild.nodeValue;	
429 					stylename=stylename.substring(1,stylename.length);
430 				}
431 				if(this.styles[stylename]){
432 					if(this.styles[stylename][pm.childNodes[j].nodeName]){
433 					}		
434 				}
435 			}
436 		}
437 		//name + description
438 		try{
439 		var name=pm.getElementsByTagName("name").item(0).firstChild.nodeValue
440 		data.properties.name=name;
441 		}catch(e){}
442 		try{
443 		var description=pm.getElementsByTagName("description").item(0).firstChild.nodeValue
444 		data.properties.description=description;
445 		}catch(e){}
446 		return data; //this.generateFeature(data);
447 	}

536 	/**
537 	 * Expects a DOMElement and extracts all "coordinates" tags
538 	 * Returns a two dimensional array of points or an array where array[0][0] is set to null 
539 	 * 
540 	 * @param {DOMElement} node
541 	 * @see http://code.google.com/intl/de/apis/kml/documentation/kmlreference.html#coordinates
542 	 * @returns Array
543 	 */
544 	this.parseCoordinates = function(node) {
545 		var coordinates = node.getElementsByTagName("coordinates");
546 		var pointArray = new Array();
547 		
548 		if(coordinates.length == 0) {
549 			// if no coordinates found, make an array of two dimensions and set the first value null
550 			// an two dimensional array is expected by the callerfunctions
551 			pointArray[0] = new Array();
552 			pointArray[0][0] = null;
553 		}
554 		
555 		for(var i = 0; i < coordinates.length; i++) {
556 			pointArray[i] = new Array(); 
557 			var coordinatesString = '';
558 			
559 			// Note: This loop is very important for Firefox!
560 			//		 Child nodes are max. 4096 characters long!
561 			//		 In contrast, Safari&Chrome put the same 60000 characters long string into 1 childNode!
562 			for(var k = 0; k < coordinates[i].childNodes.length; k++) {
563 				var value = coordinates[i].childNodes[k].nodeValue;
564 				if(value !=null && value.length > 0) {
565 					coordinatesString = coordinatesString + value.trim();
566 				}
567 			}
568 
569 			//khtml.maplib.base.Log.log('child nodes length:'+coordinates[i].childNodes.length);
570 			//khtml.maplib.base.Log.log('coordinatesString:'+coordinatesString.length);
571 			var coordinatesArray = coordinatesString.split(/\s/); // Split at any whitespace character (whitespace/newline/tab/..)
572 
573 			var j_pointArray = 0;
574 			for(var j = 0; j < coordinatesArray.length; j++) {
575 				var coordinatesString2 = coordinatesArray[j].trim();
576 				if(coordinatesString2.length > 0) {
577 					var singlePoint = coordinatesString2.split(',');
578 					pointArray[i][j_pointArray] = new khtml.maplib.LatLng(singlePoint[1]/*lat*/,singlePoint[0]/*long*/);

602 	/**
603 	 * Parse an "Icon" tag.
604 	 * 
605 	 * Parse tags: href,gx:x,gx:y,gx:w,gx:h.
606 	 * Ignore tags: refreshMode, refreshInterval, viewRefreshMode, viewRefreshTime, viewBoundScale, viewFormat, httpQuery.
607 	 * 
608 	 * @see http://code.google.com/intl/de/apis/kml/documentation/kmlreference.html#icon
609 	 * @param {DOMElement} node
610 	 */
611 	this.parseIcon = function(node) {
612 		var imgSrc = this._imgBasePath + node.getElementsByTagName("href")[0].childNodes[0].nodeValue;
613 		
614 		var myIcon = null;
615 		var myImg = document.createElement("img");
616 		myImg.setAttribute("src", imgSrc);
617 		
618 		// check if image size (gx:w/gx:h) or image position (gx:x/gx:y) tags are available
619 		if(node.getElementsByTagName("gx:x").length != 0 
620 			|| node.getElementsByTagName("gx:y").length != 0 
621 			|| node.getElementsByTagName("gx:w").length != 0 
622 			|| node.getElementsByTagName("gx:h").length != 0) {
623 			
624 			// Set image size
625 			myIcon = document.createElement("div");
626 			var w = node.getElementsByTagName("gx:w")[0].childNodes[0].nodeValue;
627 			var h = node.getElementsByTagName("gx:h")[0].childNodes[0].nodeValue;
628 			
629 			myIcon.setAttribute("style", "overflow: hidden; position: absolute; width: "+w+"px; height: "+h+"px;");
630 			myIcon.naturalWidth = w; // guard original for zoom calculations
631 			myIcon.naturalHeight = h; // guard original for zoom calculations
632 			
633 			// Set image position
634 			var x = node.getElementsByTagName("gx:x")[0].childNodes[0].nodeValue;
635 			var y = node.getElementsByTagName("gx:y")[0].childNodes[0].nodeValue;
636 			myImg.setAttribute("style", "position: absolute; left: -"+x+"px; bottom: -"+y+"px;");
637 			myImg.naturalLeft = x*(-1); // guard original for zoom calculations
638 			myImg.naturalBottom = y*(-1); // guard original for zoom calculations
639 			
640 			// append image to div
641 			myIcon.appendChild(myImg);
642 		} else {
643 			// no cropping is needed
644 			myIcon = myImg;
645 		}
646 		
647 		return myIcon;
648 	}
