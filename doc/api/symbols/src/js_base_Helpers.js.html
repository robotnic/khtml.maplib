 59 /** @function */
 60 khtml.maplib.base.helpers.contains = function(bounds1, bounds2) {
 61 	// calculates if two areas are overlapping
 62 	var rangeX = new Array(bounds1.sw().lat(), bounds1.ne().lat());
 63 	var rangeY = new Array(bounds1.sw().lng(), bounds1.ne().lng());
 64 
 65 	if( (
 66 			khtml.maplib.base.helpers.between(rangeX, bounds2.sw().lat())
 67 			|| khtml.maplib.base.helpers.between(rangeX, bounds2.ne().lat())
 68 		) 
 69 		&& (
 70 			khtml.maplib.base.helpers.between(rangeY, bounds2.sw().lng()) 
 71 			|| khtml.maplib.base.helpers.between(rangeY, bounds2.ne().lng()))
 72 		) {
 73 		// they overlap
 74 		return true;
 75 	} else {
 76 		// no overlap
 77 		return false;
 78 	}
 79 }

230 /***
231  <member name="khtml.maplib.base.helpers.eventAttach" type="static method">
232  <summary>Attach an event listener to an object.</summary>
233  <param name="o">Object whose event to attach.</param>
234  <param name="t">Type of event.</param>
235  <param name="f">Method to fire when event is raised.</param>
236  <param name="fc">Context of called method f. Defaults to object o.</param>
237  <param name="c">Use capture (not available in IE).</param>
238  <param name="*">Arguments to pass to function f.</param>
239  </member>
240  ***/
241 /** @function */
242 khtml.maplib.base.helpers.eventAttach = function(o, t, f, fc, c) {
243 	var a = (arguments.length > 5 ? $a(arguments).slice(5, arguments.length)
244 			: new Array());
245 	var fn = function(e) {
246 		a.unshift(e || window.event);
247 		return f.apply((fc ? fc : o), a);
248 	}
249 	if (o.addEventListener) {
250 		if (navigator.appName.indexOf("Netscape") == -1) {
251 			if (t == "DOMMouseScroll") {
252 				t = "mousewheel";
253 			}
254 		}
255 		if (navigator.userAgent.indexOf("Safari") != -1) {
256 			if (t == "DOMMouseScroll") {
257 				o.onmousewheel = fn;
258 			} else {
259 				o.addEventListener(t, fn, c);
260 			}
261 		} else {
262 			o.addEventListener(t, fn, c);
263 		}
264 	} else {
265 		if (t == "DOMMouseScroll") {
266 			t="onmousewheel";
267 		}else{
268 			t="on"+t;
269 		}
270 		o.attachEvent(t, fn);
271 	}
272 	return{o:o,type:t,fn:fn,c:c}
273 };

281 /**
282  * Mix in methods.
283  *
284  * @function
285  * @see http://michaux.ca/articles/transitioning-from-java-classes-to-javascript-prototypes
286  * @param {Object} target
287  * @param {Object} source
288  * @param {String/Array} props leave empty to copy all methods, pass a string or array to explicitly specify which methods should be mixed in.
289 */
290 khtml.maplib.base.helpers.mixin = function(target, source, props) {
291     if (props) {
292         if (props instanceof Array) {
293             for (var i=0; i<props.length; i++) {
294                 target[props[i]] = source[props[i]];
295             }
296         } else if (source[props]) {
297             target[props] = source[props];
298         }
299     } else {
300         for (var prop in source) {
301             target[prop] = source[prop];
302         }
303     }
304 }

408 khtml.maplib.base.helpers.parseLineString = function(pointArray) {

410 		points = new Array();
411 		var pa = pointArray.split(" ");

413 			var point = pa[i].split(",");
414 			if (point.length != 2)
415 				continue;
416 			point[0] = parseFloat(point[0]);
417 			point[1] = parseFloat(point[1]);
418 			points.push(new khtml.maplib.LatLng(point[0], point[1]));
419 		}
420 	}
421 	return points;

611 			object.style.cursor = "url('" + khtml.maplib.standardimagepath + "fist.cur'), move";
612 		else
613 			object.style.cursor = "pointer";
614     }
615 	// others
616 	else {

618 		if (object.style.cursor != "-moz-" + string) {
619 			object.style.cursor = "-webkit-" + string;		// tested on safari 5 and chrome 12
620 			if (object.style.cursor != "-webkit-" + string) {
621 				object.style.cursor = "-ms-" + string;
622 				if (object.style.cursor != "-ms-" + string) {		// missing, not sure if it works
623 					object.style.cursor = "-khtml-" + string;
624 					if (object.style.cursor != "-khtml-" + string) {		// missing, not sure if it works
625 						object.style.cursor = string;
626 						if (object.style.cursor != string){
627 							object.style.cursor = "pointer";
628 						}
629 					}
630 				}
631 			}
632 		}
633 	}
