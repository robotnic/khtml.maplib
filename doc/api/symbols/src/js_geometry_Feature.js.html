 89         this.makeBounds=function(coordinates){
 90 		if(!coordinates){
 91 			var coordinates=this.geometry.coordinates;
 92 		}
 93                 var boundsSouth = 90;
 94                 var boundsNorth = -90;
 95                 var boundsWest = 180;
 96                 var boundsEast = -180;
 97 		if(coordinates instanceof khtml.maplib.geometry.LatLng){
 98 			
 99 			if(coordinates.lat){	
100 				var lng = coordinates.lng();
101 				var lat = coordinates.lat();
102 				if (lat > boundsNorth) {
103 					boundsNorth = lat;
104 				}
105 				if (lat < boundsSouth) {
106 					boundsSouth = lat;
107 				}
108 				if (lng > boundsEast) {
109 					boundsEast = lng;
110 				}
111 				if (lng < boundsWest) {
112 					boundsWest = lng;
113 				}
114 			}
115 
116 		}else{
117 			for(var i=0; i < coordinates.length;i++){
118 				if(coordinates[i] instanceof khtml.maplib.geometry.LatLng){
119 					var lng = coordinates[i].lng();
120 					var lat = coordinates[i].lat();
121 					if (lat > boundsNorth) {
122 						boundsNorth = lat;
123 					}
124 					if (lat < boundsSouth) {
125 						boundsSouth = lat;
126 					}
127 					if (lng > boundsEast) {
128 						boundsEast = lng;
129 					}
130 					if (lng < boundsWest) {
131 						boundsWest = lng;
132 					}
133 
134 				}else{
135 					var bounds=this.makeBounds(coordinates[i]);
136 					if(bounds.sw().lat() < boundsSouth){
137 						boundsSouth=bounds.sw().lat();
138 					}
139 					if(bounds.sw().lng() < boundsWest){
140 						boundsWest=bounds.sw().lng();
141 					}
142 					if(bounds.ne().lat() > boundsNorth){
143 						boundsNorth=bounds.ne().lat();
144 					}
145 					if(bounds.ne().lng() > boundsEast){
146 						boundsEast=bounds.ne().lng();
147 					}
148 				}
149 			}
150                 }
151                 //return line bounds
152                 var sw = new khtml.maplib.LatLng(boundsSouth, boundsWest);
153                 var ne = new khtml.maplib.LatLng(boundsNorth, boundsEast);
154                 var b = new khtml.maplib.geometry.Bounds(sw, ne);
155                 return (b);
156 
157         }

178         this.render=function(){
179                 if(!that.documentElement)return;
180                 //if(!that.documentElement.map.finalDraw)return;
181                 if(!that.parentNode)that.parentNode=that.featureCollection;
182                 if(khtml.maplib.base.helpers.overlaps(that.documentElement.map.bounds(),that.bbox)){
183                         that.style.display="";
184                         switch(that.documentElement.backend){
185                                 case "svg":that.renderSVG(that);
186                                         break;
187                                 case "canvas":
188                                         that.renderCanvas(that);
189                                         break;
190                                 default:
191                                         if(console && console.log)console.log("backend unknown",that.parentNode.backend);
192 
193 
194                         }
195                 }else{
196                         that.style.display="none";
197                 }
198                 if(that.geometry.type=="Point"){
199                         that.renderMarker(that.parentNode.element);
200                 }
201         }

236 	this.doSVGPath=function(feature,coordinates,close){
237 		//if map is moving there is a delta needed. For zoom not implemented (speed)
238 		var deltaTop=0;
239 		var deltaLeft=0;
240 		var fx=1;
241 		var fy=1;
242 		if(feature.documentElement.element.finished){
243 		if(feature.documentElement.element && feature.documentElement.element.topright ){
244 			deltaTop=feature.documentElement.element.topright.y
245 			deltaLeft=feature.documentElement.element.bottomleft.x;
246 			var width=feature.documentElement.element.topright.x - deltaLeft;
247 			var height=feature.documentElement.element.bottomleft.y - deltaTop;
248 
249 			//Does not work yet
250 			var fx=width / feature.documentElement.map.size.width;
251 			var fy=height / feature.documentElement.map.size.height;
252 			/*
253 			if(old && old!=deltaTop){
254 			console.log(deltaTop,deltaLeft,fx,fy);
255 			}
256 			*/
257 			old=deltaTop;
258 
259 		}
260 		}
261 		var d="";
262 		for(var i=0;i<coordinates.length;i++){
263 			if(coordinates[i] instanceof khtml.maplib.geometry.LatLng){
264 				var xy1=this.documentElement.map.latlngToXY(coordinates[i]);
265 				var xy={"x":(xy1.x -deltaLeft),"y":(xy1.y -deltaTop)};
266 				xy.x=xy.x/fx;
267 				xy.y=xy.y/fy;
268 				if(i==0){
269 					d=" M"+xy.x+" "+xy.y;
270 				}else{
271 					d+=" L"+xy.x+" "+xy.y;
272 				}
273 				if(close && i==coordinates.length -1){
274 					d+=" z";
275 				}
276 			}else{
277 				d+=this.doSVGPath(feature,coordinates[i],close);
278 			}
279 		}	
280 		return d;
281

288 			close=true;

301 			this.element.feature=feature;

322 
323 		
324 //		if(feature.className && feature.className.baseVal){
325 			this.element.className.baseVal=feature.className.baseVal;
326 //		}	
327 		if(typeof(this.element.display)!="undefined"){
328 		this.element.style.display=this.element.display;
329 		}
330 		this.element.style.display="";
331 	}

337 	this.doCanvasPath=function(type,context,coordinates,close,depth,num){
338 		var direct=false;
339 		if(!depth)depth=0;
340 		if(!num)num=0;
341 		if(type=="Polygon" && depth >0 && num >0){
342 		//	context.globalCompositeOperation="source-out";  //sorry not working
343 		 }else{
344 			context.globalCompositeOperation="destination-over";
345 		}
346                 for(var i=0;i<coordinates.length;i++){
347                         if(coordinates[i] instanceof khtml.maplib.geometry.LatLng){
348                                 var xy=this.documentElement.map.latlngToXY(coordinates[i]);
349 				var movedXY=new Object;
350 				movedXY.x=xy.x -this.documentElement.deltaX;
351 				movedXY.y=xy.y -this.documentElement.deltaY;
352                                 if(i==0){
353 					context.moveTo(movedXY.x, movedXY.y);
354                                 }else{
355 					context.lineTo(movedXY.x, movedXY.y);
356                                 }
357                                 if(close && i==coordinates.length -1){
358 					xy=this.documentElement.map.latlngToXY(coordinates[0]);
359 					context.lineTo(xy.x, xy.y);
360                                 }
361                         }else{
362                                 this.doCanvasPath(type,context,coordinates[i],close,depth+1,i);
363                         }
364                 }

401         function reduce(origObj,depth){
402                 var good=["type","features","coordinates","style","className","baseVal","geometry","fill","stroke","strokeWidth","opacity","fillOpacity","strokeOpacity","dashArray","marker"];
403                 if(origObj instanceof khtml.maplib.geometry.LatLng){
404                         return [origObj.lng(),origObj.lat()];
405                 }
406                 if(!depth)depth=0;
407                 if(depth >15)return;
408                 var retObj=new Object;
409                 if(typeof(origObj)=="string"){
410                         return origObj;
411                 }
412                 if(origObj instanceof Array){
413                         var ar=new Array();
414                         for(var i=0;i<origObj.length;i++){
415                                 ar[i]=reduce(origObj[i],depth+1);
416                         }
417                         return ar;
418                 }
419                 if(origObj instanceof CSSStyleDeclaration){
420                         var styleProperty=new Object;
421                         for(var i=0;i<origObj.length;i++){
422                                 var name=origObj[i];
423                                 var value=origObj[origObj[i]];
424                                 styleProperty[name]=value;
425                                 //var styleProperty={origObj[i]:origObj[origObj[i]]};
426                         }
427                         return styleProperty;
428                 }
429                 for(var o in origObj){
430                         var skip=true;
431                         for(var i=0;i<good.length;i++){
432                                 if(good[i]==o ){
433                                         skip=false;
434                                         break;
435                                 }
436                         }
437                         if(skip)continue;
438                         if(typeof(origObj[o])=="string"){
439                                 retObj[o]=origObj[o];
440                         }else{
441                                 retObj[o]=reduce(origObj[o],depth+1);
442                         }
443                 }
444                 return retObj;
445         }
