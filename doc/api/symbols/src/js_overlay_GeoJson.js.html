 79 	this.render=function(force){
 80 		if(!this.featureCollection)return;
 81 
 82 		//builds an index
 83 		if(this.recurseLinksDirty){
 84 			this.recurseLink();
 85 		}
 86 
 87 		/*
 88 		Fast Rendering decisson. Do full rendering or do fast move and scale.
 89 		*/
 90 		if(!force && this.backend!='vml' && (!this.map.finalDraw  || this.map.moving )){ // && this.backend=="canvas"){
 91 				if(this.element.finished){
 92 					this.visibleElement=this.element;
 93 				}else{
 94 					if(!this.oldelement)return;
 95 					this.visibleElement=this.oldelement;
 96 				}
 97 				if(!this.visibleElement.bounds)return;
 98 				this.visibleElement.style.display="";
 99 				this.visibleElement.bottomleft=this.map.latlngToXY(this.visibleElement.bounds.sw());
100 				this.visibleElement.topright=this.map.latlngToXY(this.visibleElement.bounds.ne());
101 				var top=this.visibleElement.topright.y;
102 				var left=this.visibleElement.bottomleft.x;
103 				this.visibleElement.style.top=top+"px";
104 				this.visibleElement.style.left=left+"px";
105 				this.visibleElement.style.width=(this.visibleElement.topright.x - this.visibleElement.bottomleft.x) +"px";
106 				this.visibleElement.style.height=(-this.visibleElement.topright.y + this.visibleElement.bottomleft.y) +"px";
107 				var h=this.visibleElement.offsetLeft;  //prevend browser cheating
108 			return;
109 		}
110 		if(!this.element){
111 			//returns the rendering target
112 			this.element=this.renderElement();
113 
114 		}
115 
116 		/*
117 		rendering is done in a seperate thread. 
118 		This seperate thread is done be setTimeout tricks.
119 		This is the only way to intercept (kill) rendering if there is something.
120 		If rendering is complete, the render element will be cloned.
121 		*/
122 		if(this.element.finished){
123 			if(this.oldelement){
124 				//throw away the old, we have a new
125 				this.root.removeChild(this.oldelement);
126 			}
127 			if(this.backend=="canvas"){
128 				this.oldelement=this.element;
129 				this.element=this.renderElement();
130 			}else{
131 				this.oldelement=this.element.cloneNode(true);
132 			}
133 			this.root.appendChild(this.oldelement);
134 			this.oldelement.bounds=this.element.bounds;
135 			this.oldelement.style.display="";
136 			this.element.style.display="none";
137 			this.element.finished=false;
138 			this.element.topright=null;  //sorry tricky
139 
140 			this.element.setAttribute("viewBox","0 0 "+this.map.size.width+" "+this.map.size.height);
141 		}
142 
143 		this.element.style.top="0px";
144 		this.element.style.left="0px";
145 		//this.root.style.display="none";
146 		this.element.style.width=this.map.size.width+"px";
147 		this.element.style.height=this.map.size.height +"px";
148 		this.element.bounds=this.map.bounds();  
149 //		this.root.style.top=this.deltaY+"px";
150 //		this.root.style.left=this.deltaX+"px";
151 		this.lastzoom=this.map.zoom();
152 		this.clear();  //canvas only
153 		var base=this.featureCollection;
154 	
155 		//now the funny part starts	
156 		this.recurse(base,base);
157 		this.deltaX=0;
158 		this.deltaY=0;
159 	}

229 	this.recurse=function(feature,time){
230 		var that=this;
231 		this.busy=true;
232 		if(this.topright || (!this.map.doTheOverlays && !this.map.finalDraw)){
233 			return;
234 		}
235 		/* bremse */
236 		/* To simulate slow computers. For deveopter */
237 		/*
238 		for(var i=0;i<100000;i++){
239 			var x=document.body.offsetWidth;
240 		}
241 		*/
242 
243 
244 		/**
245 		Slow processes will be killed. The Sceduler need timing
246 		*/
247 		if(!time){
248 			var time=new Date();
249 			delta=0;
250 		}else{
251 			var now=new Date();
252 			var delta=now - time;
253 		}
254 
255 
256 		/*
257 		Render a feature an setTimout for next rendering contract.
258 		If you do not understand this part. Please, Please hands of here!!!
259 		*/
260 
261 		if(feature && feature.following ){
262 			if(delta < 50){
263 				//prevent to much recursion (stack overflow);
264 				for(var i=0;i<100;i++){
265 					if(!feature)break;
266 					feature.render();	
267 					feature=feature.following;
268 				}
269 				if(feature){
270 					//non interceptable, but faster direct function call
271 					that.recurse(feature.following,time);
272 				}else{
273 					that.cleanup();
274 					return;
275 				}
276 			}else{
277 				//killable function call
278 				var tempFunc=function(){
279 					var time=new Date();
280 					that.recurse(feature.following,time);
281 				}
282 				setTimeout(tempFunc,0);
283 			}
284 		}else{
285 			that.cleanup();
286 		}
287 		//finaly do the payload
288 		if(feature){
289 			feature.render();	
290 		}
291 	}

310 		if(!geojson)geojson=base;
311 		if(geojson.type=="FeatureCollection"){
312 			var feature=new khtml.maplib.geometry.FeatureCollection(geojson,this); //make real object from geojson

314 			var feature=new khtml.maplib.geometry.Feature(geojson,this); //make real object from geojson

316 		if(parentNode)feature.parentNode=parentNode;
317 		if(parentNode)feature.documentElement=parentNode.documentElement;
318 		if(parentNode)feature.owner=parentNode;
319 		if(geojson.type=="FeatureCollection"){
320 			if(geojson.features){
321 				for(var i=0;i<geojson.features.length;i++){
322 					var subFeature=this.recurseInit(base,geojson.features[i],feature);
323 					feature.features.push(subFeature);
324 				}
325 			}
326 		}
327 		return feature;

338 	this.recurseLink=function(base,feature,depth){
339 		if(!base){
340 			//initialize if function is called without parmeters
341 			this.removeLinks();
342 			var base=this.featureCollection;
343 			this.last=null;
344 		}
345 		if(!depth)depth=0; //for debug only
346 		if(!feature)feature=base;
347 		if(this.last){
348 			feature.preceding=this.last;
349 			this.last.following=feature;
350 		}
351 		this.last=feature;
352 		if(feature.type=="FeatureCollection"){
353 			for(var i=0;i<feature.features.length;i++){
354 				this.recurseLink(base,feature.features[i],depth+1);
355 				if(i==0){
356 					feature.features[i].precedingSibling=null;
357 				}
358 				if(i>0){
359 					feature.features[i].precedingSibling=feature.features[i -1];
360 				}
361 				if(i< feature.features.length-1){
362 					feature.features[i].followingSibling=feature.features[i +1];
363 				}
364 				if(i== feature.features.length-1){
365 					feature.features[i].followingSibling=null;
366 				}
367 			}
368 			if(feature.features[0]){
369 				feature.firstChild=feature.features[0];
370 				feature.lastChild=feature.features[feature.features.length -1];
371 			}
372 		}
373 		if(depth==0){
374 			//finished
375 			this.recurseLinksDirty=false;
376 		}
377 	}
