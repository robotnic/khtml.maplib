299 			this.startZZ = this.position.zoom;
300 			var x = (X1 + X2) / 2 / this.faktor / this.sc;
301 			var y = (Y1 + Y2) / 2 / this.faktor / this.sc;
302 			this.startMoveX = this.moveX - x;

310 	this._move = function(evt) {
311 		//khtml.maplib.base.Log.debug('this.move');
312 		
313 		if (evt.preventDefault) {
314 			evt.preventDefault(); // The W3C DOM way
315 		} else {
316 			evt.returnValue = false; // The IE way
317 		}
318 
319 		//	this.mousedownTime=null;
320 		if (evt.touches.length == 1) {
321 			if (this.moveok) {
322 				this.lastMoveX = this.moveX;
323 				this.lastMoveY = this.moveY;
324 				this.lastMoveTime = new Date(evt.timeStamp);
325 				this.moveX = this.pageX(evt.touches[0]) / this.faktor / this.sc
326 						+ this.startMoveX;
327 				this.moveY = this.pageY(evt.touches[0]) / this.faktor / this.sc
328 						+ this.startMoveY;
329 				if (!this.zoomOutStarted) {
330 					if ((Math.abs(this.moveX - this.oldMoveX) > 5)
331 							|| (Math.abs(this.moveY - this.oldMoveY) > 5)) {
332 						//alert(this.moveX+":"+this.moveY);
333 						window.clearInterval(this.zoomOutInterval);
334 						this.zoomOutSpeed = 0.01;
335 						this.mousedownTime = null;
336 					}
337 					var center = new khtml.maplib.LatLng(this.lat, this.lng);
338 					this.setCenter2(center, this.position.zoom);
339 					this.moveAnimationBlocked = false;
340 				}
341 				if ((Math.abs(this.moveX - this.oldMoveX) > 5)
342 						|| (Math.abs(this.moveY - this.oldMoveY) > 5)) {
343 					//	alert(this.moveX+":"+this.moveX);
344 					//alert(99);
345 					this.mousedownTime = null; //prevents doubleclick if map is moved already
346 				}
347 			} else {
348 				//alert("no move");
349 			}
350 		}
351 
352 		if (evt.touches.length == 2) {
353 			this.mousedownTime = null;
354 			var X1 = this.pageX(evt.touches[0]);
355 			var Y1 = this.pageY(evt.touches[0]);
356 			var X2 = this.pageX(evt.touches[1]);
357 			var Y2 = this.pageY(evt.touches[1]);
358 			var Distance = Math.sqrt(Math.pow((X2 - X1), 2)
359 					+ Math.pow((Y2 - Y1), 2));
360 			var zoomDelta = (Distance / this.startDistance);
361 			var zz = this.startZZ + zoomDelta - 1;
362 			// ewi BugFix 1: zoomDeltaOld eingefügt
363 			// sonst springt die Karte beim Erreichen der maximalen Zoomstufe
364 			if (zz > this.tileSource.maxzoom) {
365 				zz = this.tileSource.maxzoom;
366 				zoomDelta = this.zoomDeltaOld;
367 			}
368 			else
369 				this.zoomDeltaOld = zoomDelta;
370 				
371 			var x = (X1 + X2) / 2;
372 			var y = (Y1 + Y2) / 2;
373 
374 			faktor = Math.pow(2, zz);
375 			var zoomCenterDeltaX = x / faktor - this.width / 2;
376 			var zoomCenterDeltaY = y / faktor - this.height / 2;
377 			var f = Math.pow(2, zoomDelta - 1);
378 			var dx = zoomCenterDeltaX - zoomCenterDeltaX * f;
379 			var dy = zoomCenterDeltaY - zoomCenterDeltaY * f;
380 
381 			this.moveX = (x + dx) / faktor + this.startMoveX;
382 			this.moveY = (y + dy) / faktor + this.startMoveY;
383 
384 			var center = new khtml.maplib.LatLng(this.lat, this.lng);
385 			this.setCenter2(center, zz);
386 		}
387 	}

396 			evt.returnValue = false; // The IE way
397 		}
398 		window.clearInterval(this.zoomOutInterval);
399 		this.zoomOutStarted = false;
400 		this.zoomOutSpeed = 0.01;
401 
402 		if (evt.touches.length == 0) {
403 			//khtml.maplib.base.Log.debug('this.end - evt.touches.length == 0');
404 			this.moveok = true;
405 			if (this.moveAnimationMobile) {
406 				//khtml.maplib.base.Log.debug('this.end - this.moveAnimationMobile');
407 				if (this.moveAnimationBlocked == false) {
408 					/*
409 					//khtml.maplib.base.Log.debug('this.end - this.moveAnimationBlocked == false');
410 					var speedX = this.lastMoveX - this.moveX;
411 					var speedY = this.lastMoveY - this.moveY;
412 					//khtml.maplib.base.Log.debug('this.end - before clearTimeout');
413 					clearTimeout(this.animateMoveTimeout);
414 					//khtml.maplib.base.Log.debug('this.end - after clearTimeout');
415 					this.animateMove(speedX, speedY);
416 						*/
417 
418 		var now = new Date(evt.timeStamp);
419 		var timeDelta = now - this.lastMoveTime;
420                                         //speed in pixel per second
421                                         var speedX = (this.lastMoveX - this.moveX) / timeDelta *4;
422                                         var speedY = (this.lastMoveY - this.moveY) / timeDelta *4;
423                                         //console.log(speedX,speedY);
424                                         var maxSpeed = this.wheelSpeedConfig["animateMaxSpeed"];
425                                         if (speedX > maxSpeed)
426                                                 speedX = maxSpeed;
427                                         if (speedY > maxSpeed)
428                                                 speedY = maxSpeed;
429                                         if (speedX < -maxSpeed)
430                                                 speedX = -maxSpeed;
431                                         if (speedY < -maxSpeed)
432                                                 speedY = -maxSpeed;
433 
434                                         var faktor=Math.pow(2,this.zoom());
435                                         if (Math.abs(speedX) > this.wheelSpeedConfig["animateMinSpeed"]/faktor
436                                                         || Math.abs(speedY) > this.wheelSpeedConfig["animateMinSpeed"]/faktor) {
437                                                 this._animateMove(speedX, speedY,faktor);
438                                         } else {
439                                                 //this.renderOverlays();
440                                         }
441                                 }
442 			}
443 		}
444 		
445 		// ewi BugFix 2: wenn vorher zwei Finger waren und dann nur mehr einer
446 		// sonst springt die Karte beim Auslassen eines Fingers
447 		if (evt.touches.length == 1) {
448 			this.startMoveX = this.moveX - evt.touches[0].pageX / this.faktor / this.sc;
449             this.startMoveY = this.moveY - evt.touches[0].pageY / this.faktor / this.sc;
450 		}
451 	}
